local _kfsFileCache = {}
local function execKfsPathFor(actorName, variant)
    return ZORG_ROOT .. "/anims/" .. actorName .. "/exec_" .. (variant or "default") .. ".rbxm"
end
local function hasKfsFile(actorName, variant)
    if not actorName then return false end
    variant = variant or "default"
    -- 👽 FIX: префикс "exec:" во избежание коллизий с victim-кэшем
    -- (victim использует "victim:actorName:variant" — если actorName == "victim",
    -- ключи без префикса бы пересеклись).
    local key = "exec:" .. actorName .. ":" .. variant
    if _kfsFileCache[key] ~= nil then return _kfsFileCache[key] end
    local exists = isfile and isfile(execKfsPathFor(actorName, variant)) or false
    _kfsFileCache[key] = exists
    return exists
end

end

local function hasAnyKfsFile(actorName)
    if not actorName then return false end
    local seen = {}
    for _, info in pairs(_activeExecKillerIds) do
        local v = info.variant or "default"
        if not seen[v] then
            seen[v] = true
            if hasKfsFile(actorName, v) then return true end
        end
    end
    return false
end

-- Возвращает звуки прошлого киллера к их оригинальным SoundId. Должно вызываться
-- ПЕРЕД заполнением _activeSoundMap новыми мапингами, иначе мы не сможем восстановить
-- оригиналы (info будет указывать уже на ID нового киллера).
local function clearActiveSoundMappings()
    for snd, _ in pairs(allSounds) do
        if origIds[snd] and snd.Parent then
            pcall(function() snd.SoundId = origIds[snd] end)
        end
    end
    table.clear(allSounds); table.clear(fakeIds); table.clear(origIds)
end

local function getKillerConfig(actorInfo)
    if actorInfo.Config then return actorInfo.Config end
    local folder = actorInfo.Folder
    if not folder then return nil end
    local cfgModule = folder:FindFirstChild("Config")
    if not cfgModule then return nil end
    local ok, cfg = pcall(require, cfgModule)
    if not ok then warn("ZORG👽: pcall require Config failed:", cfg); return nil end
    return cfg
end

-- 👽 V7.69: phase-helpers для loadKillerConfig. Раньше функция была монолитом ~170 строк
-- с 7-8 разными фазами, каждая писала в свои _active*-таблицы. Разбили на отдельные
-- register*/load* функции с одной ответственностью. Поведение идентично — порядок вызовов
-- сохранён, потому что между фазами есть зависимости (см. ниже).
--
-- ПОРЯДОК (важен!):
--   1. resetActiveMaps       — clearActiveSoundMappings + table.clear всех _active*
--   2. registerTerrorRadius  → заполняет _activeChaseMap
--   3. registerVoicelines    → заполняет _activeVoicelineIds
--   4. registerExecVariants  → заполняет _activeExecKillerIds, _activeExecPlayerIds,
--                              _activeLMSAnimMap (для SummonPizzaDelivery)
--   5. applyExecAnimFallbacks → читает _activeExec*Ids → пишет _activeLMSAnimMap
--                              (default victim anim + EXEC_ANIM_FALLBACK_ID для variant'ов
--                               без KFS). ЗАВИСИТ от шага 4.
--   6. registerLMSSfxOverrides → заполняет _activeLMSSfxMap + _activeFootstep1/2
--                                (c00lkidd-only через ACTOR_REGISTRY.hasLMSSfxOverrides)
--   7. loadSfxFiles          → читает cfg.Sounds → пишет _activeSoundMap (custom assets)
--   8. loadChaseFiles        → читает _activeChaseMap → пишет _activeSoundMap.
--                              ЗАВИСИТ от шага 2.
local function resetActiveMaps()
    -- Звуковые таблицы (allSounds/origIds/fakeIds): возвращаем предыдущие звуки к
    -- оригиналам, ПОТОМ чистим. Если сначала очистить, потеряем origIds и не сможем
    -- вернуть подменённые звуки к их игровым ID.
    clearActiveSoundMappings()
    table.clear(_activeChaseMap); table.clear(_activeVoicelineIds); table.clear(_activeLMSSfxMap)
    table.clear(_activeSoundMap)
    table.clear(_activeExecKillerIds); table.clear(_activeExecPlayerIds); table.clear(_activeLMSAnimMap)
    -- 👽 V7.68: сбрасываем footstep-указатели — они держат прямые ссылки на LMS_FOOTSTEP_CUSTOM
    -- из конфига предыдущего киллера. Без сброса при смене актёра остаются протухшие значения.
    _activeFootstep1, _activeFootstep2 = nil, nil
end

-- 👽 NOTE: единый union-лукап радиус → имя файла. Файл резолвится относительно
-- папки текущего актёра (Zorg/sounds/<actor>/chase/<file>), поэтому одно имя
-- "layer1.ogg" не вызывает коллизий между киллерами — у каждого своя папка.
-- Радиусы сгруппированы по актёрам:
--   c00lkidd: 60 / 40 / 20  → layer1 / layer2 / layer3
--   Slasher:  45 / 30 / 15  → layer1 / layer2 / layer3
--   общий:    6             → chase
-- registerTerrorRadius читает только те радиусы, что присутствуют в cfg.Sounds.
-- TerrorRadiusThemes конкретного киллера, поэтому "лишние" строки тихо игнорятся.
local TERROR_RADIUS_LAYER_MAP = {
    [60] = "layer1.ogg", [40] = "layer2.ogg", [20] = "layer3.ogg",  -- c00lkidd
    [45] = "layer1.ogg", [30] = "layer2.ogg", [15] = "layer3.ogg",  -- Slasher
    [6]  = "chase.ogg",
}
local function registerTerrorRadius(trt)
    if not trt then return end
    for radius, info in pairs(trt) do
        if info and info.ID then
            local raw = tostring(info.ID):match("%d+")
            if raw then
                local file = TERROR_RADIUS_LAYER_MAP[radius] or ("layer_dist" .. tostring(radius) .. ".ogg")
                _activeChaseMap[raw] = file
            end
        end
    end
end

local function registerVoicelines(voicelines)
    if not voicelines then return end
    for category, list in pairs(voicelines) do
        if type(list) == "table" then
            for _, assetUri in ipairs(list) do
                local raw = tostring(assetUri):match("%d+")
                if raw then _activeVoicelineIds[raw] = category end
            end
        end
    end
end

-- 👽 V7.64: generic-loop по всем ключам "Execution*" в Config.Animations. Любой
-- киллер с любым набором execution-вариантов поддерживается автоматически.
-- Примеры variant'ов в живых конфигах: default (Execution), 007n7, guest1337.
local function registerExecVariants(actorName, anims)
    if not anims then return end
    for animKey, animData in pairs(anims) do
        if typeof(animData) == "table" and animKey:sub(1, 9) == "Execution" then
            -- variant: "Execution" → "default", "Execution007n7" → "007n7"
            local variant = (animKey == "Execution") and "default" or animKey:sub(10):lower()
            local krRaw = tostring(animData.KillerRig or ""):match("%d+")
            local prRaw = tostring(animData.PlayerRig or ""):match("%d+")
            local dur = getExecDuration(actorName, variant)
            if krRaw then _activeExecKillerIds[krRaw] = {duration = dur, variant = variant} end
            if prRaw then _activeExecPlayerIds[prRaw] = {duration = dur, variant = variant} end
        end
    end
    if anims.SummonPizzaDelivery then
        local raw = tostring(anims.SummonPizzaDelivery):match("%d+")
        if raw then _activeLMSAnimMap[raw] = "rbxassetid://86758834842420" end
    end
end

local function applyExecAnimFallbacks(actorName)
    -- Все player-rig execution-анимации мапятся на стандартный victim-anim 135641783638394.
    -- Если для конкретного actor+variant есть собственный victim_*.rbxm, attachVictimKfsHandler
    -- перехватит его и сыграет через KFS-плеер. Если нет — играет стандартная.
    for rawId, _ in pairs(_activeExecPlayerIds) do _activeLMSAnimMap[rawId] = "rbxassetid://135641783638394" end
    -- Per-variant fallback: если для конкретного варианта (default/007n7) нет KFS-файла,
    -- эта exec-анимация заменяется на стандартный fallback. Каждый вариант независим.
    for rawId, info in pairs(_activeExecKillerIds) do
        if not hasKfsFile(actorName, info.variant) then
            _activeLMSAnimMap[rawId] = EXEC_ANIM_FALLBACK_ID
        end
    end
end

-- 👽 V7.68: LMS-SFX подмена применяется ТОЛЬКО для c00lkidd. Asset'ы из LMS_SFX_CUSTOM_MAP
-- и LMS_FOOTSTEP_CUSTOM были подобраны под него. Для не-c00lkidd карта остаётся пустой →
-- patchSound в activateLMSSfx пропускает → играют оригинальные игровые звуки.
local function registerLMSSfxOverrides(actorName, sounds)
    if not sounds then return end
    local entry = getActorEntry(actorName)
    if not (entry and entry.hasLMSSfxOverrides) then return end

    if sounds.Footsteps then
        local raw1 = tostring(sounds.Footsteps[1] or ""):match("%d+")
        local raw2 = tostring(sounds.Footsteps[2] or ""):match("%d+")
        if raw1 then _activeLMSSfxMap[raw1] = LMS_FOOTSTEP_CUSTOM[1]; _activeFootstep1 = LMS_FOOTSTEP_CUSTOM[1] end
        if raw2 then _activeLMSSfxMap[raw2] = LMS_FOOTSTEP_CUSTOM[2]; _activeFootstep2 = LMS_FOOTSTEP_CUSTOM[2] end
    end

    for cfgKey, customUri in pairs(LMS_SFX_CUSTOM_MAP) do
        local gameUri = sounds[cfgKey]
        if gameUri then
            local raw = tostring(gameUri):match("%d+")
            if raw and not _activeLMSSfxMap[raw] then _activeLMSSfxMap[raw] = customUri end
        end
    end
end

-- Грузит файлы Zorg/sounds/<actor>/sfx/<Key>.ogg для каждого ключа cfg.Sounds (не table) +
-- Zorg/sounds/<actor>/sfx/Footstep<N>.ogg для каждого элемента cfg.Sounds.Footsteps. Чтения
-- проходят через pcall(getcustomasset). Файлы которых нет — игнорируются (silently).
local function loadSfxFiles(actorName, sounds)
    if not sounds then return end
    local sfxDir = soundDirFor(actorName, "sfx")
    local function loadOne(raw, path)
        if not raw then return end
        if not (isfile and isfile(path)) then return end
        local ok, asset = pcall(getcustomasset, path)
        if ok and asset then _activeSoundMap[raw] = asset end
    end
    for key, value in pairs(sounds) do
        if not SFX_KEYS_BLACKLIST[key] and typeof(value) ~= "table" then
            local raw = tostring(value):match("%d+")
            loadOne(raw, sfxDir .. "/" .. key .. ".ogg")
        end
    end
    if sounds.Footsteps and typeof(sounds.Footsteps) == "table" then
        for i, v in ipairs(sounds.Footsteps) do
            local raw = tostring(v):match("%d+")
            loadOne(raw, sfxDir .. "/Footstep" .. i .. ".ogg")
        end
    end
end

local function loadChaseFiles(actorName)
    local chaseDir = soundDirFor(actorName, "chase")
    for id, file in pairs(_activeChaseMap) do
        local path = chaseDir .. "/" .. file
        if isfile and isfile(path) then
            local ok, asset = pcall(getcustomasset, path)
            if ok and asset then _activeSoundMap[id] = asset end
        end
    end
end

local function loadKillerConfig(actorInfo)
    local cfg = getKillerConfig(actorInfo)
    if not cfg then warn("ZORG👽: Не удалось загрузить Config для", actorInfo.ActorName); return false end

    local actorName = actorInfo.ActorName
    resetActiveMaps()
    registerTerrorRadius(cfg.Sounds and cfg.Sounds.TerrorRadiusThemes)

    applyExecAnimFallbacks(actorName)
    registerLMSSfxOverrides(actorName, cfg.Sounds)
    loadSfxFiles(actorName, cfg.Sounds)
    loadChaseFiles(actorName)
    return true
end

local function shortArg(v, depth)
    depth = depth or 0
    if depth > SNIFFER_MAX_ARG_DEPTH then return "<...>" end
    local t = typeof(v)
    if t == "nil" then return "nil"
    elseif t == "boolean" then return tostring(v)
    elseif t == "number" then return tostring(v)
    elseif t == "string" then if #v > 40 then return ('"' .. v:sub(1, 40) .. '..."') else return ('"' .. v .. '"') end
    elseif t == "Vector3" then return ("V3(%.1f,%.1f,%.1f)"):format(v.X, v.Y, v.Z)
    elseif t == "CFrame" then return ("CFr(%.1f,%.1f,%.1f)"):format(v.Position.X, v.Position.Y, v.Position.Z)
    elseif t == "Instance" then return ("[%s:%s]"):format(v.ClassName, v.Name)
    elseif t == "buffer" then return ("<buf:%d>"):format(buffer.len(v))
    elseif t == "table" then
        local parts = {}; local n = 0
        for k, v2 in pairs(v) do
            n = n + 1
            if n > 5 then table.insert(parts, "...") break end
            table.insert(parts, tostring(k) .. "=" .. shortArg(v2, depth + 1))
        end
        return "{" .. table.concat(parts, ",") .. "}"
    else return ("<" .. t .. ">") end
end

local function snifferLog(name, args)
    if not SNIFFER_ENABLED then return end
    local nameStr = tostring(name)
    if not SNIFFER_VERBOSE and SNIFFER_SKIP_NAMES[nameStr] then return end
    local first = not _seenEventNames[nameStr]
    _seenEventNames[nameStr] = (_seenEventNames[nameStr] or 0) + 1
    if not first and _seenEventNames[nameStr] % 50 ~= 0 then return end
    local parts = {}
    for i = 1, math.min(#args, 4) do table.insert(parts, shortArg(args[i])) end
    if #args > 4 then table.insert(parts, ("...+%d"):format(#args - 4)) end
    local tag = first and "[NEW]" or ("[x" .. tostring(_seenEventNames[nameStr]) .. "]")
    print(("ZORG SNIFF %s %s(%s)"):format(tag, nameStr, table.concat(parts, ", ")))
end

local function dispatchToListeners(name, args)
    local list = _eventListeners[name]
    if not list then return end
    for _, cb in ipairs(list) do pcall(cb, unpack(args)) end
end

local EVENT_CODE_NAMES = {
    [1] = "UpdateCharacterPosition", [2] = "FootSP", [3] = "StatusApplied",
    [4] = "StatusUpdated", [5] = "StatusRemoved", [6] = "REPLICATE_DEBUG", [7] = "DEVELOPER_REPLICATION_TIME",
}

local function onNetworkEvent(name, packedOrBuffer)
    local resolved = name
    if type(name) == "number" then
        resolved = EVENT_CODE_NAMES[name] or ("CODE_" .. tostring(name))
    end
    -- 👽 V7.91 PERF: если нет слушателей И sniffer выключен — выходим до распаковки.
    -- UpdateCharacterPosition приходит каждый кадр; лишний alloc + итерация по пустому
    -- списку накапливаются. Распаковываем args только когда они реально нужны.
    if not SNIFFER_ENABLED and not _eventListeners[resolved] then return end
    local args
    if typeof(packedOrBuffer) == "table" then
        local n = packedOrBuffer.n or #packedOrBuffer
        args = {}
        for i = 1, n do args[i] = packedOrBuffer[i] end
    elseif typeof(packedOrBuffer) == "buffer" then
        args = {packedOrBuffer}
    else
        args = {packedOrBuffer}
    end
    snifferLog(resolved, args)
    dispatchToListeners(resolved, args)
end

    local remote = netModule:WaitForChild("RemoteEvent", 10)
    if not remote then return end
    _networkRemote = remote
    track(remote.OnClientEvent:Connect(onNetworkEvent))
    local uremote = netModule:FindFirstChild("UnreliableRemoteEvent")
    if uremote then track(uremote.OnClientEvent:Connect(onNetworkEvent)) end
end

local _soundCache = {}
local function loadSoundsFromDir(dir)
    if _soundCache[dir] then return _soundCache[dir] end
    _soundCache[dir] = {}
    if not (isfolder and isfolder(dir)) then return _soundCache[dir] end
    local ok, files = pcall(listfiles, dir)
    if not ok then return _soundCache[dir] end
    for _, p in ipairs(files) do
        -- 👽 FIX: Lua-патерны не поддерживают | как alternation, а () — это capture,
        -- не группировка. Старый паттерн "%.(mp3|ogg|wav)$" буквально требовал, чтобы
        -- после точки шли символы m,p,3,|,o,g,g,|,w,a,v подряд — никогда не матчилось,
        -- и весь папочный механизм войслайнов/звуков с диска был мёртв.
        local ext = p:lower():match("%.([^%.\\/]+)$")
        if ext == "mp3" or ext == "ogg" or ext == "wav" then
            local ok2, a = pcall(getcustomasset, p)
            if ok2 then table.insert(_soundCache[dir], a) end
        end
    end
    return _soundCache[dir]
end

local function loadVoicelineSounds(actorName, category)

    local list = loadSoundsFromDir(soundDirFor(actorName, "voicelines/" .. category))
    if #list > 0 then return list end
    return loadSoundsFromDir(soundDirFor(actorName, "voicelines"))
end

local function silenceVoiceline(snd)
    local id = tostring(snd.SoundId or ""):match("%d+")
    if not id or not _activeVoicelineIds[id] then return end
    if not snd:GetAttribute("ZORG_VL_VOL") then snd:SetAttribute("ZORG_VL_VOL", snd.Volume) end
    snd.Volume = 0; pcall(function() snd:Stop() end)
    local c1 = snd:GetPropertyChangedSignal("IsPlaying"):Connect(function()
        if _voicelinesBlocked and snd.IsPlaying then snd.Volume = 0; pcall(function() snd:Stop() end) end
    end)
    local c2 = snd:GetPropertyChangedSignal("Volume"):Connect(function()
        if _voicelinesBlocked and snd.Volume > 0 then snd.Volume = 0 end
    end)
    -- 👽 V7.92 SCOPE PHASE 1: коннекты идут в _voicelineScope. При unblockVoicelines scope
    -- разрушается и пересоздаётся — нет накопления Disconnect'ed-записей.
    _voicelineScope:Add(c1); _voicelineScope:Add(c2)
end

infectVoiceline = function(snd)
    if _voicelinesBlocked or not masterEnabled then return end
    if not (typeof(snd) == "Instance" and snd:IsA("Sound")) then return end
    local id = tostring(snd.SoundId or ""):match("%d+")
    if not id or not _activeVoicelineIds[id] then return end
    local category = _activeVoicelineIds[id]
    local actorName = _currentKiller and _currentKiller.ActorName
    local s = loadVoicelineSounds(actorName, category)
    if #s == 0 then return end
    _voicelineIds[snd] = snd.SoundId
    snd.SoundId = s[math.random(1, #s)]
end

blockVoicelines = function()
    if _voicelinesBlocked then return end
    _voicelinesBlocked = true
    local function scan(parent)
        for _, d in ipairs(parent:GetDescendants()) do if d:IsA("Sound") then silenceVoiceline(d) end end
    end
    if getinstances then
        for _, o in pairs(getinstances()) do if typeof(o) == "Instance" and o:IsA("Sound") then silenceVoiceline(o) end end
    else
        scan(Workspace); scan(game:GetService("SoundService")); scan(RS); scan(game:GetService("CoreGui")); scan(Players)
    end
    for snd, orig in pairs(_voicelineIds) do if snd.Parent then snd.SoundId = orig end end
    table.clear(_voicelineIds)
    for _, svc in ipairs({Workspace, game:GetService("SoundService"), RS, game:GetService("CoreGui"), Players}) do
        -- 👽 V7.92 SCOPE PHASE 1: было table.insert(_vlConns, svc.DescendantAdded:Connect(...)).
        -- Scope:Connect — идиоматичный шорткат для signal:Connect + Add в один вызов.
        _voicelineScope:Connect(svc.DescendantAdded, function(d)
            if _voicelinesBlocked and d:IsA("Sound") then silenceVoiceline(d) end
        end)
    end
end

unblockVoicelines = function()
    if not _voicelinesBlocked then return end
    _voicelinesBlocked = false
    -- 👽 V7.92 SCOPE PHASE 1 (FIX 5): destroy + пересоздание scope. Без пересоздания
    -- следующий blockVoicelines пытался бы добавлять в destroyed-scope (тот моментально
    -- cleanup'ил бы добавляемое). Legacy-цикл по _vlConns удалён в Фазе 5.
    _voicelineScope:Destroy()
    _voicelineScope = _zorgScope:Child("voicelines")
    if getinstances then
        for _, o in pairs(getinstances()) do
            if typeof(o) == "Instance" and o:IsA("Sound") then
                local vol = o:GetAttribute("ZORG_VL_VOL")
                if vol then o.Volume = vol; o:SetAttribute("ZORG_VL_VOL", nil) end
                infectVoiceline(o)
            end
        end
    end
end

local function loadKFSGeneric(cache, missing, pathFn, actorName, variant)
    if not actorName then return nil end
    variant = variant or "default"
    cache[actorName]   = cache[actorName]   or {}
    missing[actorName] = missing[actorName] or {}
    if cache[actorName][variant]   then return cache[actorName][variant] end
    if missing[actorName][variant] then return nil end
    local path = pathFn(actorName, variant)
    if not (isfile and isfile(path)) then
        missing[actorName][variant] = true
        return nil
    end
    local okA, asset = pcall(getcustomasset, path)
    if not okA then
        missing[actorName][variant] = true
        return nil
    end
    local okO, objs = pcall(game.GetObjects, game, asset)
    if not okO or not objs or not objs[1] then
        missing[actorName][variant] = true
        return nil
    end
    local kfs = objs[1]
    if kfs.ClassName ~= "KeyframeSequence" then
        for _, d in ipairs(kfs:GetDescendants()) do
            if d:IsA("KeyframeSequence") then kfs = d; break end
        end
    end
    if kfs.ClassName ~= "KeyframeSequence" then
        kfs:Destroy()
        missing[actorName][variant] = true
        return nil
    end
    kfs.Parent = RS
    cache[actorName][variant] = kfs
    return kfs
end

local _customExecRunning = {}

local function collectMotors(rig)
    local motors = {}
    for _, d in ipairs(rig:GetDescendants()) do
        if d:IsA("Motor6D") and d.Part1 then
            motors[d.Part1.Name] = d
        end
    end
    return motors
end

local function collectPoses(parent, acc)
    acc = acc or {}
    for _, c in ipairs(parent:GetChildren()) do
        if c:IsA("Pose") then
            acc[c.Name] = c
            collectPoses(c, acc)
        end
    end
    return acc
end

local POSE_IGNORE = { HumanoidRootPart = true, RootJoint = true }

-- 👽 V7.67 PERF: предобработка KFS вынесена сюда и кэшируется в loadExecKFS/loadVictimKFS.
-- Раньше каждый запуск казни тратил ~1500 итераций (для KFS на ~100 keyframes × 15 костей)
-- на построение bonePoses + N сортировок. Это детерминированная pure-функция данных KFS,
-- результат у одного и того же файла всегда одинаков → кэш безопасен.
-- Возвращает {bonePoses = {[boneName] = {{time, cf}, ...}}, totalLength = number}.
local function buildBonePosesFromKFS(kfs)
    local keyframes = kfs:GetKeyframes()
    if #keyframes == 0 then return nil end
    table.sort(keyframes, function(a, b) return a.Time < b.Time end)
    local totalLength = keyframes[#keyframes].Time

    local bonePoses = {}
    for _, kf in ipairs(keyframes) do
        local poses = collectPoses(kf)
        local kfTime = kf.Time
        for boneName, pose in pairs(poses) do
            if not POSE_IGNORE[boneName] then
                -- pose.Weight: задокументированное свойство Pose. Pcall раньше был паранойей,
                -- удалён вместе с переходом на pre-build (вызывается один раз при load).
                if pose.Weight ~= 0 then
                    if not bonePoses[boneName] then bonePoses[boneName] = {} end
                    table.insert(bonePoses[boneName], {time = kfTime, cf = pose.CFrame})
                end
            end
        end
    end
    for _, list in pairs(bonePoses) do
        table.sort(list, function(a, b) return a.time < b.time end)
    end

    return {bonePoses = bonePoses, totalLength = totalLength}
end

-- Weak-map: ключ — KFS-instance, значение — pre-built данные. При уничтожении KFS
-- (например, в cleanupAll) запись автоматически GC'тся.
local _kfsPrebuiltCache = setmetatable({}, {__mode = "k"})
local function getKFSPrebuilt(kfs)
    if not kfs then return nil end
    local cached = _kfsPrebuiltCache[kfs]
    if cached then return cached end
    local built = buildBonePosesFromKFS(kfs)
    if built then _kfsPrebuiltCache[kfs] = built end
    return built
end

local function playKFSOnRig(kfs, rig, humanoid)
    if _customExecRunning[rig] then return false, "already running" end
    _customExecRunning[rig] = true

    -- 👽 V7.67 PERF: pre-built кэш. Если KFS уже разбирался ранее — переиспользуем готовые
    -- bonePoses/totalLength. Иначе buildBonePosesFromKFS вызовется один раз и закэшируется.
    local prebuilt = getKFSPrebuilt(kfs)
    if not prebuilt then
        _customExecRunning[rig] = nil
        return false, "no keyframes in KFS"
    end
    local bonePoses = prebuilt.bonePoses
    local totalLength = prebuilt.totalLength

    local motors = collectMotors(rig)

    -- 👽 V7.59: Никаких манипуляций с треками или Animator'ом.
    -- Мы привязываемся к RunService.PreSimulation. Это происходит после Animator
    -- и перед Physics. Движок пишет в Transform свой Idle, а мы его тут же 
    -- перекрываем. Animate.lua полностью счастлив и ничего не замечает.

    -- 👽 V7.67: elapsed накапливается из dt самой PreSimulation вместо tick().
    -- tick() формально deprecated в Roblox; os.clock() — это CPU-time, не wall-clock
    -- (стопается на hiccup'ах). Накопление dt из самого callback'а: (a) совпадает с тем
    -- что видит движок; (b) не зависит от внешних часов; (c) никогда не deprecates.
    local elapsed = 0
    local fadeElapsed = 0
    local stopped = false
    local missingFrames = 0
    local stepConnection = nil

    -- 👽 V7.91 PERF: per-bone cursor для O(1) амортизированного поиска кадра.
    -- elapsed монотонно растёт → индекс никогда не идёт назад в штатном режиме.
    -- Курсоры сбрасываются только в beginFadeOut (однократно при завершении анимации).
    local boneCursors = {}
    for boneName, poseList in pairs(bonePoses) do
        boneCursors[boneName] = 1
    end

    local FADE_ENABLED = true
    local STOP_FADE_DURATION = 0.35 
    local fading = false
    local fadeStartPoses = nil
    local FADE_IGNORE = {RootJoint = true}

    local function stop()
        if stopped then return end
        stopped = true
        if stepConnection then stepConnection:Disconnect() end
        _customExecRunning[rig] = nil
        zlog("CFG", "[ExecAnim] Плеер остановлен на " .. rig.Name .. ". Гармония сохранена без единого шва.")
    end
    
    local function beginFadeOut(currentElapsed)
        if stopped or fading then return end
        if not FADE_ENABLED then stop(); return end

        fadeStartPoses = {}
        local clampedForFade = math.min(currentElapsed or elapsed, totalLength - 0.001)

        -- Делаем математический слепок нашей финальной позы KFS
        for boneName, poseList in pairs(bonePoses) do
            if not FADE_IGNORE[boneName] and #poseList > 0 then
                local targetCF
                local prev, nxt
                for i = 1, #poseList - 1 do
                    if poseList[i].time <= clampedForFade and poseList[i+1].time > clampedForFade then
                        prev = poseList[i]; nxt = poseList[i+1]; break
                    end
                end
                if prev and nxt then
                    local span = nxt.time - prev.time
                    local alpha = span > 0 and (clampedForFade - prev.time) / span or 0
                    targetCF = prev.cf:Lerp(nxt.cf, alpha)
                elseif clampedForFade <= poseList[1].time then
                    targetCF = poseList[1].cf
                else
                    targetCF = poseList[#poseList].cf
                end
                fadeStartPoses[boneName] = targetCF
            end
        end

        fading = true
        fadeElapsed = 0
        zlog("CFG", "[ExecAnim] начат ЖИВОЙ fade-out в оригинальный engineCF на " .. rig.Name)
    end

    local function stepFn(dt)
        local rigOk = rig and rig.Parent ~= nil
        local humOk = humanoid and humanoid.Parent ~= nil
        if not (rigOk and humOk) then
            missingFrames = missingFrames + 1
            if missingFrames > 30 then stop() return end
            return
        end
        missingFrames = 0

        elapsed = elapsed + dt

        -- Если время вышло и fade ещё не начат — стартуем его ПРЯМО СЕЙЧАС (в этом же кадре),
        -- чтобы не пропустить ни одного кадра записи в motor.Transform. Если пропустить —
        -- Animator успеет показать свою idle-позу, и игрок увидит teleport на долю секунды.
        if elapsed >= totalLength and not fading then
            beginFadeOut(elapsed)
            -- НЕ делаем return: проваливаемся ниже в fade-блок, чтобы записать первый кадр fade.
        end

        -- 👽 ЖИВОЕ СГЛАЖИВАНИЕ В ФОНОВЫЙ ТРЕК
        if fading then
            fadeElapsed = fadeElapsed + dt
            local alpha = math.min(fadeElapsed / STOP_FADE_DURATION, 1)
            local eased = alpha * alpha * (3 - 2 * alpha)
            
            for boneName, startCF in pairs(fadeStartPoses) do
                local motor = motors[boneName]
                if motor and motor.Parent then
                    -- Читаем живой Transform: Animator только что записал сюда idle.
                    -- Лерпим нашу финальную KFS-позу в их idle.
                    -- 👽 V7.67: pcall удалён — присваивание Motor6D.Transform на живом
                    -- (motor.Parent проверен) инстансе не бросает; pcall был паранойей.
                    local engineCF = motor.Transform
                    motor.Transform = startCF:Lerp(engineCF, eased)
                end
            end
            
            if alpha >= 1 then stop() end
            return
        end

        local clampedElapsed = math.min(elapsed, totalLength - 0.001)

        -- 👽 ЖЕСТКОЕ ПЕРЕКРЫТИЕ НА СТАДИИ PreSimulation
        -- 👽 V7.91 PERF: курсор-based поиск кадра — O(1) амортизированно.
        -- elapsed монотонно растёт → курсор продвигается вперёд, никогда не назад.
        for boneName, poseList in pairs(bonePoses) do
            local motor = motors[boneName]
            if motor and motor.Parent then
                -- Продвигаем курсор вперёд пока следующий keyframe ещё не наступил
                local cur = boneCursors[boneName] or 1
                while cur < #poseList and poseList[cur + 1].time <= clampedElapsed do
                    cur = cur + 1
                end
                boneCursors[boneName] = cur

                local targetCF
                if cur < #poseList then
                    local prev, nxt = poseList[cur], poseList[cur + 1]
                    local span = nxt.time - prev.time
                    local alpha = span > 0 and (clampedElapsed - prev.time) / span or 0
                    targetCF = prev.cf:Lerp(nxt.cf, alpha)
                elseif #poseList > 0 then
                    if clampedElapsed <= poseList[1].time then
                        targetCF = poseList[1].cf
                    else
                        targetCF = poseList[#poseList].cf
                    end
                end

                if targetCF then
                    motor.Transform = targetCF
                end
            end
        end
    end

    -- ИСПОЛЬЗУЕМ PreSimulation. Это происходит ПОСЛЕ Animator и ДО физики.
    -- (PreSimulation = современный алиас старого RunService.Stepped, который deprecated.)
    -- dt-параметр приходит от движка → используется для накопления elapsed.
    stepConnection = RunService.PreSimulation:Connect(stepFn)
    return true, beginFadeOut
end

        pcall(beginFadeOut)
    end)
    return true
end

    -- 👽 V7.92 SCOPE PHASE 2: per-execution scope. Создаётся ДО hidePart — благодаря этому
    -- hidePart использует execScope напрямую без forward-declare трюка с execTrack_fwd.
    -- Все коннекты, syncConn, watcher'ы, ragdoll-mimic'и идут сюда. На завершении казни
    -- (или early-return) — execScope:Destroy() чистит всё в LIFO. unhideVictimParts —
    -- отдельная логика (восстановление Transparency на оригинальной жертве), вызывается
    -- параллельно со Destroy на путях где жертва должна остаться видимой.
    local execScope = _zorgScope:Child("vexec_" .. victimModel.Name)

    -- Подготавливаем структуру hiddenParts заранее — будет заполнена при скрытии.
    -- Каждая запись: hiddenParts[part] = {origTransparency = N}.
    -- Нужно сохранить оригинальные значения ДО клонирования, чтобы потом восстановить
    -- их на клоне (иначе клон унаследует Transparency=1 от скрытого оригинала).
    local hiddenParts = {}

    -- 1. ВНАЧАЛЕ клонируем (пока оригинал ещё видим — клон унаследует правильный Transparency).
    local clone = victimModel:Clone()
    clone.Name = victimModel.Name .. "_ZORG_VCLONE"
    for _, d in ipairs(clone:GetDescendants()) do
        if d:IsA("Script") or d:IsA("LocalScript") then
            pcall(function() d:Destroy() end)
        elseif d:IsA("BasePart") then
            d.CanCollide = false; d.CanQuery = false; d.CanTouch = false; d.Anchored = false
            d.LocalTransparencyModifier = 0
        end
    end
    local cloneRoot = clone:FindFirstChild("HumanoidRootPart")
    if not cloneRoot then
        pcall(function() clone:Destroy() end)
        -- 👽 V7.92 SCOPE PHASE 2: execScope создан выше — чистим. hiddenParts пустой
        -- (hidePart ещё не вызывался), unhideVictimParts не нужен.
        execScope:Destroy()
        return false
    end

    -- 2. ТЕПЕРЬ скрываем оригинал.
    -- LocalTransparencyModifier работает только для собственной модели LocalPlayer'а.
    -- Для других игроков нужен Transparency. Применяем оба для надёжности.
    --
    -- 👽 FIX: вместо опроса всех hiddenParts каждый RenderStepped — навешиваем
    -- GetPropertyChangedSignal("Transparency") / ("LocalTransparencyModifier")
    -- на каждую часть. В steady state коннекты молчат; срабатывают только когда
    -- движок реально дёргает значение назад. На 30-50 частях × 2 сигнала это
    -- ~60-100 idle коннектов, но без работы каждый кадр.
    --
    -- 👽 V7.92 SCOPE PHASE 2: execTrack_fwd forward-declare убран — execScope создан
    -- ВЫШЕ (до hidePart), поэтому hidePart использует его напрямую. Compensating loop
    -- "перенести коннекты, созданные до execTrack_fwd, в execConns" тоже исчез как
    -- неактуальный. hiddenParts[d].conns остаётся как маркер "часть уже скрыта" (см.
    -- проверку `if not hiddenParts[d].conns then` ниже) — это логика, не cleanup.
    local function hidePart(d)
        local isPart = d:IsA("BasePart")
        local isVisual = d:IsA("Decal") or d:IsA("Texture")
        if not (isPart or isVisual) then return end
        if not hiddenParts[d] then
            hiddenParts[d] = {origTransparency = d.Transparency}
        end
        pcall(function()
            d.LocalTransparencyModifier = 1
            d.Transparency = 1
        end)
        -- Реактивные re-enforcer'ы: если кто-то снаружи (Animate script, сервер,
        -- ещё один скрипт) перетёр значение — мгновенно ставим назад.
        if not hiddenParts[d].conns then
            hiddenParts[d].conns = {}
            local c1 = d:GetPropertyChangedSignal("Transparency"):Connect(function()
                if d.Transparency ~= 1 then pcall(function() d.Transparency = 1 end) end
            end)
            table.insert(hiddenParts[d].conns, c1)
            execScope:Add(c1)
            if isPart then
                local c2 = d:GetPropertyChangedSignal("LocalTransparencyModifier"):Connect(function()
                    if d.LocalTransparencyModifier ~= 1 then
                        pcall(function() d.LocalTransparencyModifier = 1 end)
                    end
                end)
                table.insert(hiddenParts[d].conns, c2)
                execScope:Add(c2)
            end
        end
    end
    for _, d in ipairs(victimModel:GetDescendants()) do hidePart(d) end

    -- 3. Anchor + offset относительно киллера (наш, не серверный)
    cloneRoot.Anchored = true
    local function applyOffset()
        if killerRoot and killerRoot.Parent then
            cloneRoot.CFrame = killerRoot.CFrame * getVictimOffset(actorName, variant)
        end
    end
    applyOffset()

    -- Копируем состояние Motor6D из оригинала в клон (для корректного стартового состояния)
    local origMotors = {}
    for _, d in ipairs(victimModel:GetDescendants()) do
        if d:IsA("Motor6D") then origMotors[d.Name] = {C0 = d.C0, C1 = d.C1, Transform = d.Transform} end
    end
    for _, d in ipairs(clone:GetDescendants()) do
        if d:IsA("Motor6D") then
            local saved = origMotors[d.Name]
            if saved then pcall(function() d.C0 = saved.C0; d.C1 = saved.C1; d.Transform = saved.Transform end) end
        end
    end

    clone.Parent = Workspace

    -- 👽 V7.92 SCOPE PHASE 2: блок создания execConns/execTrack/cleanExecConns +
    -- compensating loop (перенос коннектов hidePart в execConns) — удалён.
    -- execScope создан ВЫШЕ (до hidePart), hidePart уже клал коннекты в него напрямую.

    -- Sync позиции клона: каждый кадр обновляем offset относительно киллера.
    -- Это нужно если киллер двигается во время казни — клон жертвы движется с ним.
    local syncConn
    syncConn = RunService.RenderStepped:Connect(function()
        if not (clone.Parent and killerRoot and killerRoot.Parent and cloneRoot and cloneRoot.Parent) then
            if syncConn then syncConn:Disconnect() end
            return
        end
        cloneRoot.CFrame = killerRoot.CFrame * getVictimOffset(actorName, variant)
    end)
    execScope:Add(syncConn)

    -- 👽 FIX: RenderStepped-цикл удалён. Раньше каждый кадр итерировались все
    -- hiddenParts (30-50 BasePart'ов + Decal'ы) с двумя сравнениями на каждую.
    -- Теперь Transparency и LocalTransparencyModifier охраняются реактивно через
    -- GetPropertyChangedSignal в hidePart — никакой работы в steady state.

    -- Watcher новых descendant'ов оригинала: если игра во время казни добавит accessory
    -- или другую часть на жертву — её тоже надо скрыть, иначе видим "плавающий зонтик".
    execScope:Connect(victimModel.DescendantAdded, function(d) hidePart(d) end)

    -- Запрещаем Humanoid'у клона дёргаться
    local cloneHum = clone:FindFirstChildOfClass("Humanoid")
    if cloneHum then
        cloneHum.WalkSpeed = 0; cloneHum.JumpPower = 0; cloneHum.AutoRotate = false
        pcall(function() cloneHum:ChangeState(Enum.HumanoidStateType.Physics) end)
    end
    -- Стопаем любые анимации, которые могли начаться на клоне
    local cloneAnimator = cloneHum and cloneHum:FindFirstChildOfClass("Animator")
    if cloneAnimator then
        for _, t in ipairs(cloneAnimator:GetPlayingAnimationTracks()) do pcall(function() t:Stop(0) end) end
    end

    -- 4. Играем KFS на клоне через тот же playKFSOnRig
    -- 👽 FIX: helper для восстановления оригинальных значений Transparency/LTM.
    -- Раньше при early-return восстанавливался только LocalTransparencyModifier=0,
    -- а Transparency оставался = 1 → жертва оставалась невидимой у других игроков.
    -- Теперь используем сохранённое hiddenParts[d].origTransparency.
    local function unhideVictimParts()
        for d, saved in pairs(hiddenParts) do
            if d and d.Parent then
                pcall(function()
                    d.LocalTransparencyModifier = 0
                    if saved and saved.origTransparency ~= nil then
                        d.Transparency = saved.origTransparency
                    end
                end)
            end
        end
    end

    local ok, beginFadeOut = playKFSOnRig(kfs, clone, cloneHum or victimHum)
    if not ok then
        execScope:Destroy()
        pcall(function() clone:Destroy() end)
        unhideVictimParts()
        return false
    end

    local dur = execDuration or EXEC_DURATION

    -- 5. Через `dur` секунд — fade-out KFS-анимации и переход в ragdoll-финал
    task.delay(dur, function()
        pcall(beginFadeOut)

        -- Ждём окончания fade-out перед ragdoll'ом (STOP_FADE_DURATION = 0.35 в playKFSOnRig)
        task.wait(0.4)
        if not clone.Parent then
            execScope:Destroy()
            unhideVictimParts()
            return
        end

        -- Отключаем sync — клон теперь живёт сам по физике
        if syncConn then syncConn:Disconnect(); syncConn = nil end

        -- Готовим Humanoid клона к ragdoll'у
        if cloneHum then
            pcall(function()
                cloneHum:ChangeState(Enum.HumanoidStateType.Physics)
                cloneHum.PlatformStand = true; cloneHum.AutoRotate = false
                cloneHum.WalkSpeed = 0; cloneHum.JumpPower = 0
                for _, st in pairs(Enum.HumanoidStateType:GetEnumItems()) do
                    if st ~= Enum.HumanoidStateType.Physics and st ~= Enum.HumanoidStateType.None then
                        pcall(function() cloneHum:SetStateEnabled(st, false) end)
                    end
                end
            end)
        end
        task.wait()

        -- Unanchor всех частей + восстановление collision (по образцу playerCloneExecution)
        forEachPart(clone, function(d)
            pcall(function()
                d.Anchored = false; d.CanCollide = true; d.Massless = false; d.CollisionGroup = "Default"
            end)
        end)

        -- Уничтожаем Hip-моторы (ноги отваливаются) — точная копия из playerCloneExecution
        local LIMB_MOTORS = {["Right Hip"] = true, ["Left Hip"] = true}
        for _, d in ipairs(clone:GetDescendants()) do
            if d:IsA("Motor6D") and LIMB_MOTORS[d.Name] then pcall(function() d:Destroy() end) end
        end

        -- HRP больше не нужен — удаляем
        local hrp = clone:FindFirstChild("HumanoidRootPart")
        if hrp then pcall(function() hrp:Destroy() end) end

        -- BallSocket-суставы для Neck, Right Shoulder, Left Shoulder
        local function makeRagdollJoint(motor)
            local part0, part1 = motor.Part0, motor.Part1
            if not (part0 and part1) then return end
            local a0 = Instance.new("Attachment"); a0.CFrame = motor.C0; a0.Parent = part0
            local a1 = Instance.new("Attachment"); a1.CFrame = motor.C1; a1.Parent = part1
            local bs = Instance.new("BallSocketConstraint")
            bs.Attachment0 = a0; bs.Attachment1 = a1; bs.LimitsEnabled = true; bs.UpperAngle = 60
            bs.TwistLimitsEnabled = true; bs.TwistLowerAngle = -45; bs.TwistUpperAngle = 45; bs.Parent = part0
            pcall(function() motor:Destroy() end)
        end
        local RAGDOLL_JOINTS = {["Neck"] = true, ["Right Shoulder"] = true, ["Left Shoulder"] = true}
        for _, d in ipairs(clone:GetDescendants()) do
            if d:IsA("Motor6D") and RAGDOLL_JOINTS[d.Name] then pcall(function() makeRagdollJoint(d) end) end
        end

        -- Подменяем серверный ragdoll: прячем его, копируем его constraints на наш клон.
        -- Это делает финальную позу клона физически корректной (без серверного ragdoll'а
        -- клон бы просто лежал статично; теперь его части связаны с серверным ragdoll'ом
        -- и движутся как он). Точная копия логики из playerCloneExecution.
        task.spawn(function()
            local ragdollsFold = Workspace:FindFirstChild("Ragdolls")
            if not ragdollsFold then return end
            local origName = victimModel.Name
            local origRagdoll; local waited = 0
            while not origRagdoll and waited < 0.5 do
                origRagdoll = ragdollsFold:FindFirstChild(origName)
                if not origRagdoll then task.wait(0.05); waited = waited + 0.05 end
            end
            if not (origRagdoll and clone.Parent) then return end
            local function hideRagdoll(model)
                for _, d in ipairs(model:GetDescendants()) do
                    if d:IsA("BasePart") or d:IsA("Decal") or d:IsA("Texture") then
                        pcall(function() d.LocalTransparencyModifier = 1 end)
                    end
                end
            end
            hideRagdoll(origRagdoll)
            execScope:Connect(origRagdoll.DescendantAdded, function(d)
                if d:IsA("BasePart") or d:IsA("Decal") or d:IsA("Texture") then
                    pcall(function() d.LocalTransparencyModifier = 1 end)
                end
            end)
            local RAGDOLL_LIMBS = {Head = true, Torso = true, ["Right Arm"] = true, ["Left Arm"] = true}
            for _, d in ipairs(clone:GetDescendants()) do
                local isUnnamedLimbAttachment = d:IsA("Attachment") and d.Name == "" and RAGDOLL_LIMBS[d.Parent.Name]
                if d:IsA("BallSocketConstraint") or isUnnamedLimbAttachment then pcall(function() d:Destroy() end) end
            end
            for _, partName in ipairs({"Head", "Torso", "Right Arm", "Left Arm"}) do
                local origPart = origRagdoll:FindFirstChild(partName)
                local clonePart = clone:FindFirstChild(partName)
                if origPart and clonePart then
                    for _, c in ipairs(origPart:GetChildren()) do
                        if c:IsA("Attachment") or c:IsA("Constraint") then
                            local copy = c:Clone(); copy.Parent = clonePart
                        end
                    end
                end
            end
        end)

        -- Через 6 секунд — финальная очистка клона
        task.delay(6, function()
            execScope:Destroy()
            pcall(function() clone:Destroy() end)
            -- Не восстанавливаем видимость оригинала — он уже мёртв и заменён ragdoll'ом,
            -- который мы спрятали выше. Сервер сам уничтожит оригинал в Players folder.
        end)
    end)

    return true
end

local mt = getrawmetatable(game)
local oldIndex, oldNewIndex = mt.__index, mt.__newindex

local function rollLMS()
    -- 👽 V7.71: pick может быть false (sentinel для "оригинальной музыки") или string.
    -- `pick or nil` нормализует false→nil; остальной код (resolveLMSAsset и др.) ждёт
    -- именно nil как "не подменять".
    local pick = LMS_OPTIONS[math.random(1, #LMS_OPTIONS)]
    currentRandomLMS = pick or nil
end

local function hasMilestoneSkin(m)
    for _, pl in ipairs(Players:GetPlayers()) do
        local c = pl.Character
        if (c and c.Name == m.Name) or pl.Name == m.Name then
            local pd = pl:FindFirstChild("PlayerData")
            local equipped = pd and pd:FindFirstChild("Equipped")
            local skins = equipped and equipped:FindFirstChild("Skins")
            if skins then
                for _, sv in ipairs(skins:GetChildren()) do
                    local v = tostring(sv.Value)
                    if v:match("^Milestone75") or v:match("^Milestone100") then return true end
                end
            end
        end
    end
    return false
end

local function checkMilestone()
    if DEBUG_MILESTONE then return true end
    local now = tick()
    if now - _milestoneCache.ts < MILESTONE_CACHE_TTL then return _milestoneCache.result end
    local pf = Workspace:FindFirstChild("Players")
    if not pf then _milestoneCache = {ts = now, result = false}; return false end
    local kf, sf = pf:FindFirstChild("Killers"), pf:FindFirstChild("Survivors")
    if not (kf and sf) then _milestoneCache = {ts = now, result = false}; return false end
    local killers = kf:GetChildren()
    if #killers == 0 then _milestoneCache = {ts = now, result = false}; return false end
    local alive = {}
    for _, s in ipairs(sf:GetChildren()) do local h = s:FindFirstChild("Humanoid") if h and h.Health > 0 then table.insert(alive, s) end end
    local res = #alive == 1 and hasMilestoneSkin(alive[1]) and hasMilestoneSkin(killers[1])
    _milestoneCache = {ts = now, result = res}
    return res
end

local function resolveLMSDir()
    -- 👽 V7.70: убран c00lkidd-fallback. Раньше: если у текущего киллера нет
    -- Zorg/sounds/<actor>/lms/ — функция возвращала Zorg/sounds/c00lkidd/lms/, и для
    -- любого нового киллера в LMS играла c00lkidd-овская музыка. Это шло вразрез с
    -- решением v7.68, где LMS-SFX строго ограничили c00lkidd через ACTOR_REGISTRY.
    -- Теперь логика симметричная: нет своей папки → возвращаем nil → resolveLMSAsset
    -- возвращает nil → tryHandleLMSMusicSet/infectSound не подменяют → играет оригинальная
    -- игровая LMS-музыка. Для c00lkidd fallback и так был бесполезен (смотрел в ту же
    -- папку что и основной путь) — его поведение не изменилось.
    if _currentKiller and _currentKiller.ActorName then
        local d = soundDirFor(_currentKiller.ActorName, "lms")
        if isfolder and isfolder(d) then return d, "current(" .. (_currentKiller.ActorName) .. ")" end
    end
    return nil, "nil"
end

local function resolveLMSAsset()
    local target = checkMilestone() and LMS_MILESTONE or currentRandomLMS
    if not target then return nil, "target=nil" end
    local dir, src = resolveLMSDir()
    if not dir then return nil, "dir=nil" end
    local a = getAsset(dir, target)
    if not a then return nil, "file not found" end
    return a, "OK"
end

local function infectSound(snd)
    if typeof(snd) ~= "Instance" or not snd:IsA("Sound") then return end
    -- 👽 V7.91 FIX: masterEnabled проверяется здесь (пассивный скан), как это уже
    -- делается на hot-path в __newindex (строка installHook). Раньше LMS-ветка infectSound
    -- подменяла SoundId даже когда мастер выключен (кнопка "КАСТОМ ЗВУКИ: ВЫКЛ").
    if not masterEnabled then return end
    local raw = snd.SoundId:match("%d+")
    if not raw then return end
    if raw == LMS_ID then
        allSounds[snd] = raw; origIds[snd] = snd.SoundId
        local a, msg = resolveLMSAsset()
        if a then
            fakeIds[snd] = snd.SoundId; snd.SoundId = a; pcall(function() snd.Looped = false end)
        end
    elseif _activeSoundMap[raw] then
        fakeIds[snd] = snd.SoundId; origIds[snd] = snd.SoundId; allSounds[snd] = raw
        pcall(function() snd.SoundId = _activeSoundMap[raw] end)
    end
end

local function disableMaster()
    for snd, _ in pairs(allSounds) do
        if origIds[snd] then
            pcall(function() fakeIds[snd] = nil; snd.SoundId = origIds[snd] end)
        end
    end
end

local function enableMaster()
    -- 👽 V7.69 FIX: раньше enableMaster пропускал LMS_ID. Если пользователь во время LMS
    -- дёргал кнопку «КАСТОМ ЗВУКИ»: ВЫКЛ → ВКЛ — звуки возвращались, но LMS-музыка оставалась
    -- оригинальной до её собственной перезагрузки. Теперь LMS-музыка восстанавливается
    -- отдельной веткой через resolveLMSAsset (берёт текущий рандом / milestone target).
    for snd, raw in pairs(allSounds) do
        if raw == LMS_ID then
            local a = resolveLMSAsset()
            if a then
                pcall(function() fakeIds[snd] = origIds[snd]; snd.SoundId = a end)
            end
        elseif _activeSoundMap[raw] then
            pcall(function() fakeIds[snd] = origIds[snd]; snd.SoundId = _activeSoundMap[raw] end)
        end
    end
end

local function tryHandleVoicelineSet(t, k, v, raw)
    if not _activeVoicelineIds[raw] or _voicelinesBlocked then return false end
    local category = _activeVoicelineIds[raw]
    local actorName = _currentKiller and _currentKiller.ActorName
    local sounds = loadVoicelineSounds(actorName, category)
    if #sounds == 0 then return false end
    local chosen = sounds[math.random(1, #sounds)]
    _voicelineIds[t] = v
    oldNewIndex(t, k, chosen)
    return true
end

local function tryHandleSoundMap(t, k, v, raw)
    if raw == LMS_ID then return false end
    local asset = _activeSoundMap[raw]
    if not asset then return false end
    fakeIds[t] = v; origIds[t] = v; allSounds[t] = raw
    oldNewIndex(t, k, asset)
    return true
end

local function tryHandleLMSMusicSet(t, k, v, raw)
    if raw ~= LMS_ID then return false end
    origIds[t] = v; allSounds[t] = raw
    local asset, msg = resolveLMSAsset()
    if asset then
        fakeIds[t] = v; pcall(function() oldNewIndex(t, "Looped", false) end)
        oldNewIndex(t, k, asset)
        return true
    end
    return false
end

local function tryHandleAnimationSet(t, k, v, raw)
    if _lmsSfxActive and _activeLMSAnimMap[raw] then
        oldNewIndex(t, k, _activeLMSAnimMap[raw])
        return true
    end
    return false
end

local function installHook()
    setreadonly(mt, false)
    mt.__newindex = newcclosure(function(t, k, v)
        if checkcaller() or typeof(t) ~= "Instance" then return oldNewIndex(t, k, v) end
        if t:IsA("Sound") and k == "SoundId" and masterEnabled then
            local raw = tostring(v):match("%d+")
            if raw then
                if tryHandleVoicelineSet(t, k, v, raw) then return end
                if tryHandleSoundMap(t, k, v, raw)     then return end
                if tryHandleLMSMusicSet(t, k, v, raw)  then return end
            end
        elseif t:IsA("Animation") and k == "AnimationId" then
            local raw = tostring(v):match("%d+")
            if raw and tryHandleAnimationSet(t, k, v, raw) then return end
        end
        return oldNewIndex(t, k, v)
    end)
    mt.__index = newcclosure(function(t, k)
        if k ~= "SoundId" then return oldIndex(t, k) end
        if checkcaller() or typeof(t) ~= "Instance" or not t:IsA("Sound") then return oldIndex(t, k) end
        if _voicelineIds[t] then return _voicelineIds[t] end
        if fakeIds[t] then return fakeIds[t] end
        return oldIndex(t, k)
    end)
    setreadonly(mt, true)
end

local function startMaster()
    rollLMS()
    if getinstances then for _, o in pairs(getinstances()) do infectSound(o); infectVoiceline(o) end end
    for _, svc in ipairs({Workspace, game:GetService("SoundService"), LP, RS, game:GetService("CoreGui")}) do
        track(svc.DescendantAdded:Connect(function(o) if masterEnabled then infectSound(o); infectVoiceline(o) end end))
    end
    track(Workspace.DescendantAdded:Connect(function(o)
        if o.Name == "Ingame" and o.Parent and o.Parent.Name == "Map" and masterEnabled then
            rollLMS(); deactivateLMSSfx()
            _milestoneCache = {ts = 0, result = false}
            for snd, raw in pairs(allSounds) do if raw == LMS_ID then pcall(function() fakeIds[snd] = nil; snd.SoundId = origIds[snd] end); allSounds[snd], origIds[snd] = nil, nil end end
            task.wait(1)
            if getinstances then for _, s in pairs(getinstances()) do infectSound(s); infectVoiceline(s) end end
        end

local function applyParts(skinFolder, partMap, rig, rotateMap, hideList, target)
    local clones = {}
    if not (skinFolder and rig) then return clones end
    for myName, rigName in pairs(partMap) do
        local myPart, rigPart = skinFolder:FindFirstChild(myName), rig:FindFirstChild(rigName)
        if myPart and rigPart then
            local c = myPart:Clone()
            c.Anchored, c.CanCollide, c.CanQuery, c.CanTouch = false, false, false, false
            c.Massless = true; c.Parent = target
            c.CFrame = (rotateMap and rotateMap[rigName]) and (rigPart.CFrame * CFrame.Angles(0, math.rad(180), 0)) or rigPart.CFrame
            local w = Instance.new("WeldConstraint"); w.Part0, w.Part1, w.Parent = rigPart, c, rigPart
            -- 👽 FIX: сохраняем оригинальный Transparency перед скрытием, чтобы cleanup
            -- мог его восстановить. Используем attribute (a) чтобы не плодить глобальные
            -- таблицы; (b) attribute переживёт re-application без double-saved (проверка ниже).
            if rigPart:GetAttribute("ZORG_ORIG_TRANSPARENCY") == nil then
                rigPart:SetAttribute("ZORG_ORIG_TRANSPARENCY", rigPart.Transparency)
            end
            rigPart.Transparency = 1
            _cloned[target] = _cloned[target] or {}; table.insert(_cloned[target], c); table.insert(clones, c)
        end
    end
    if hideList then
        for _, name in ipairs(hideList) do
            for _, obj in ipairs(rig:GetDescendants()) do
                if obj.Name == name and obj:IsA("BasePart") then
                    if obj:GetAttribute("ZORG_ORIG_TRANSPARENCY") == nil then
                        obj:SetAttribute("ZORG_ORIG_TRANSPARENCY", obj.Transparency)
                    end
                    obj.Transparency = 1
                end
            end
        end
    end
    return clones
end

-- 👽 FIX: восстановление оригинального Transparency на rig-частях после удаления скина.
-- Проходит по дескендантам model и для каждой части, у которой есть attribute
-- ZORG_ORIG_TRANSPARENCY (поставленный applyParts), возвращает Transparency на исходное
-- значение и снимает attribute. Безопасно вызывать на уничтоженных моделях (no-op).
local function restoreRigTransparency(model)
    if not model or not model.Parent then return end
    for _, d in ipairs(model:GetDescendants()) do
        if d:IsA("BasePart") then
            local orig = d:GetAttribute("ZORG_ORIG_TRANSPARENCY")
            if orig ~= nil then
                pcall(function() d.Transparency = orig end)
                d:SetAttribute("ZORG_ORIG_TRANSPARENCY", nil)
            end
        end
    end
end

local function findBoneFuzzy(rig, hint)
    if not (rig and hint) then return nil end
    local b = rig:FindFirstChild(hint)
    if b and b:IsA("BasePart") then return b end
    b = rig:FindFirstChild(hint .. "c")
    if b and b:IsA("BasePart") then return b end
    for _, prefix in ipairs({"Upper", "Lower"}) do
        b = rig:FindFirstChild(prefix .. hint)
        if b and b:IsA("BasePart") then return b end
    end
    local hlow = hint:lower()
    for _, d in ipairs(rig:GetDescendants()) do
        if d:IsA("BasePart") and d.Name:lower():find(hlow, 1, true) then return d end
    end
    return nil
end

local function getAccessoryRoot(acc)
    if acc:IsA("BasePart") then return acc end
    if acc:IsA("Model") then
        if acc.PrimaryPart then return acc.PrimaryPart end
        for _, d in ipairs(acc:GetDescendants()) do if d:IsA("BasePart") then return d end end
    end
    if acc:IsA("Accessory") then
        local h = acc:FindFirstChild("Handle")
        if h and h:IsA("BasePart") then return h end
        for _, d in ipairs(acc:GetDescendants()) do if d:IsA("BasePart") then return d end end
    end
    return nil
end

local function applyAccessories(skinFolder, rig, target, referenceRig, isMinion)
    if not skinFolder then return end
    for _, acc in ipairs(skinFolder:GetChildren()) do
        if (acc:IsA("BasePart") or acc:IsA("Model") or acc:IsA("Accessory")) then
            local boneHintObj = acc:FindFirstChild("_BoneHint")
            local boneName = boneHintObj and boneHintObj.Value or (isMinion and ACCESSORY_MAP[acc.Name])
            if boneName then
                local rigPart = findBoneFuzzy(rig, boneName)
                if rigPart then
                    local offsetObj = acc:FindFirstChild("_Offset")
                    local offset = CFrame.new()
                    if offsetObj then
                        offset = offsetObj.Value
                    elseif referenceRig then
                        local refBone = findBoneFuzzy(referenceRig, boneName)
                        if refBone then
                            local accRoot = getAccessoryRoot(acc)
                            if accRoot then offset = refBone.CFrame:ToObjectSpace(accRoot.CFrame) end
                        end
                    end

                    local clone = acc:Clone()
                    for _, n in ipairs({"_BoneHint", "_Offset"}) do
                        local x = clone:FindFirstChild(n); if x then x:Destroy() end
                    end
                    for _, d in ipairs(clone:GetDescendants()) do
                        if d:IsA("Weld") or d:IsA("AccessoryWeld") then d:Destroy() end
                    end

                    local cloneRoot = getAccessoryRoot(clone)
                    if cloneRoot then
                        local function makeNonPhysical(p)
                            p.Anchored = false; p.CanCollide = false
                            p.CanQuery = false; p.CanTouch = false; p.Massless = true
                        end
                        if clone:IsA("BasePart") then makeNonPhysical(clone) end
                        for _, d in ipairs(clone:GetDescendants()) do if d:IsA("BasePart") then makeNonPhysical(d) end end

                        if clone:IsA("Model") then clone:PivotTo(rigPart.CFrame * offset) else cloneRoot.CFrame = rigPart.CFrame * offset end
                        clone.Parent = target

                        local w = Instance.new("WeldConstraint")
                        w.Part0, w.Part1, w.Parent = rigPart, cloneRoot, rigPart

                        if clone:IsA("Model") then
                            for _, d in ipairs(clone:GetDescendants()) do
                                if d:IsA("BasePart") and d ~= cloneRoot then
                                    local hasWeld = false
                                    for _, ch in ipairs(d:GetChildren()) do
                                        if ch:IsA("WeldConstraint") or ch:IsA("Weld") then hasWeld = true; break end
                                    end
                                    if not hasWeld then
                                        local w2 = Instance.new("WeldConstraint"); w2.Part0, w2.Part1, w2.Parent = cloneRoot, d, cloneRoot
                                    end
                                end
                            end
                        end
                        _cloned[target] = _cloned[target] or {}; table.insert(_cloned[target], clone)
                    end
                end
            end
        end
    end
end

local function smoothLerp(cur, tgt, dt, sm) return cur:Lerp(tgt, 1 - math.exp(-dt / math.max(sm, 0.001))) end

        local hum = km:FindFirstChildOfClass("Humanoid")
        local anim = hum and hum:FindFirstChildOfClass("Animator")
        if not anim then return end

        lmsTrack(anim.AnimationPlayed:Connect(function(trk)
            if not _lmsSfxActive then return end
            local a = trk.Animation
            if not a then return end
            local raw = tostring(a.AnimationId or ""):match("%d+")
            if not raw or not _activeLMSAnimMap[raw] then return end
            if _activeExecPlayerIds[raw] then return end

                local ok, t = pcall(function() return anim:LoadAnimation(na) end)
                if ok and t then t.Priority = trk.Priority; rep[raw] = t end
            end
            if rep[raw] then pcall(function() rep[raw]:Play(0, wt or 1, sp or 1) end) end
        end))
    end

    local kf2 = pf2 and pf2:FindFirstChild("Killers")
    local sf2 = pf2 and pf2:FindFirstChild("Survivors")

        for _, km in ipairs(kf2:GetChildren()) do attachAnimWatcher(km) end

    end
    if sf2 then
        for _, sm in ipairs(sf2:GetChildren()) do attachAnimWatcher(sm) end
        lmsTrack(sf2.ChildAdded:Connect(function(sm) if _lmsSfxActive then attachAnimWatcher(sm) end end))

            if n:FindFirstChildOfClass("Humanoid") then attachAnimWatcher(n) end
        end
        lmsTrack(ingameNpcsFold.ChildAdded:Connect(function(n)
            if _lmsSfxActive and n:FindFirstChildOfClass("Humanoid") then attachAnimWatcher(n) end
        end))
    end

    local function patchSound(obj)
        if not _lmsSfxActive then return end
        if not (typeof(obj) == "Instance" and obj:IsA("Sound")) then return end
        if obj.Name:match("^ZORG_PHANTOM") then return end
        local raw = tostring(obj.SoundId or ""):match("%d+")
        if not raw or not _activeLMSSfxMap[raw] then return end
        local base = obj.Volume > 0.05 and obj.Volume or 0.85
        obj:SetAttribute("ZORG_ORIG_VOLUME", base)
        pcall(function() obj.Volume = 0 end)
        lmsTrack(obj:GetPropertyChangedSignal("Volume"):Connect(function()
            if not _lmsSfxActive then return end
            if obj.Volume > 0 then base = obj.Volume; obj:SetAttribute("ZORG_ORIG_VOLUME", base); obj.Volume = 0 end
        end))
        local proxy = Instance.new("Sound")
        proxy.Name = "ZORG_PHANTOM_" .. raw; proxy.SoundId = _activeLMSSfxMap[raw]; proxy.Volume = base
        for _, pr in ipairs{"PlaybackSpeed", "RollOffMinDistance", "RollOffMaxDistance", "RollOffMode", "Looped"} do pcall(function() proxy[pr] = obj[pr] end) end
        proxy.Parent = obj.Parent
        if obj.IsPlaying then pcall(function() proxy.TimePosition = obj.TimePosition; proxy:Play() end) end
        lmsTrack(obj:GetPropertyChangedSignal("IsPlaying"):Connect(function()
            if not _lmsSfxActive then return end
            if obj.IsPlaying then pcall(function() proxy.TimePosition, proxy.Volume = obj.TimePosition, base; proxy:Play() end) else pcall(function() proxy:Stop() end) end
        end))
        lmsTrack(obj:GetPropertyChangedSignal("PlaybackSpeed"):Connect(function() pcall(function() proxy.PlaybackSpeed = obj.PlaybackSpeed end) end))
        lmsTrack(obj.Destroying:Connect(function() pcall(function() proxy:Destroy() end) end))
    end

    local function attach(km)
        if not km then return end
        for _, d in ipairs(km:GetDescendants()) do patchSound(d) end
        lmsTrack(km.DescendantAdded:Connect(patchSound))
    end

    local kf = getKillersFolder()
    if not kf then return end
    for _, km in ipairs(kf:GetChildren()) do attach(km) end
    lmsTrack(kf.ChildAdded:Connect(function(km) if _lmsSfxActive then attach(km) end end))
end

        for _, trk in pairs(cache) do pcall(function() trk:Stop(0); trk:Destroy() end) end
    end
    table.clear(_animReplacedCaches)
    if _killerAnimState then
        pcall(function() _killerAnimState.idle:Stop(0); _killerAnimState.idle:Destroy() end)
        pcall(function() _killerAnimState.move:Stop(0); _killerAnimState.move:Destroy() end)

    end
    local kf = getKillersFolder()

            local eased = alpha * alpha * (3 - 2 * alpha)
            for p, orig in pairs(origTransparency) do if p.Parent then p.Transparency = orig + (1 - orig) * eased end end
            if alpha >= 1 then
                conn:Disconnect(); pcall(function() jane:Destroy() end)
                for i = #_ghosts, 1, -1 do if _ghosts[i] == jane then table.remove(_ghosts, i); break end end
                applyLMSSkin(killerModel)
            end
        end))
    end

    local function processVictimExecution(victimModel, va, trk, raw, motorDefaults, vRep)
        if _victimExecRunning[victimModel] then return end
        _victimExecRunning[victimModel] = true
        local vroot = victimModel:FindFirstChild("HumanoidRootPart") or victimModel.PrimaryPart
        if vroot then
            _pendingPlayerRig[victimModel] = {tick(), vroot}
            -- 👽 FIX: страховочная очистка на случай, если findVictim не подберёт эту запись
            -- (например, ghost ещё не успел вызвать playGhostExecution). TTL = 0.3s,
            -- но даём запас в 1s чтобы точно не помешать живому find'у.
            task.delay(1.0, function() _pendingPlayerRig[victimModel] = nil end)
        end
        local newId = _activeLMSAnimMap[raw]
        if not newId then task.delay(5, function() _victimExecRunning[victimModel] = nil end); return end
        local wt = trk.WeightTarget
        local execInfo = _activeExecPlayerIds[raw]
        local duration = execInfo and execInfo.duration or EXEC_DURATION
        local isPlayer = Players:GetPlayerFromCharacter(victimModel) ~= nil
        if isPlayer then
            playerCloneExecution(victimModel, vroot, newId, wt, duration)
            task.delay(8, function() _victimExecRunning[victimModel] = nil end)
            return
        end
        for _, t in pairs(va:GetPlayingAnimationTracks()) do pcall(function() t:Stop(0) end) end
        if motorDefaults then
            for motor, defs in pairs(motorDefaults) do
                if motor and motor.Parent then pcall(function() motor.C0 = defs.C0; motor.C1 = defs.C1 end) end
            end
        end
        task.wait()
        if not vRep[raw] then
            local na = Instance.new("Animation"); na.AnimationId = newId
            local ok, t = pcall(function() return va:LoadAnimation(na) end)
            if ok and t then
                local prio = Enum.AnimationPriority.Action4 or Enum.AnimationPriority.Action
                t.Priority = prio; vRep[raw] = t
            end
        end
        if vRep[raw] then
            local newTrk = vRep[raw]
            pcall(function() newTrk:Play(0, wt or 1, 1) end)
        end
        task.delay(5, function() _victimExecRunning[victimModel] = nil end)
    end

    -- который через :Connect(victim.Destroying, scope.Destroy) автоматически чистит scope
    -- когда жертва уничтожена игрой. _victimScopes — weak-keyed map для лукапа существующего
    -- scope при рекурсивном вызове (когда ветка А wait-Humanoid снова зовёт watchVictimAnimator).
    -- weak-keys чтобы entry не держал victimModel в жизни (та может быть уничтожена игрой).
    local _victimScopes = setmetatable({}, {__mode = "k"})

    local function watchVictimAnimator(victimModel)
        -- Получаем существующий scope или создаём новый. Рекурсивный вызов после wait-Humanoid
        -- попадёт сюда повторно и переиспользует уже-созданный scope.
        local vScope = _victimScopes[victimModel]
        if not vScope then
            vScope = _c00Scope:Child("victim_" .. victimModel.Name)
            vScope:BindToInstance(victimModel)
            _victimScopes[victimModel] = vScope
            -- Cleanup-функция: убираем себя из _victimScopes при destroy (иначе при
            -- рекурсивном вызове после смерти жертвы _victimScopes[vm] вернёт dead-scope,
            -- любой :Add на нём моментально cleanup'нется и потеряется).
            vScope:Cleanup(function() _victimScopes[victimModel] = nil end)
        end

        local vh = victimModel:FindFirstChildOfClass("Humanoid")
        if not vh then
            local conn
            conn = victimModel.ChildAdded:Connect(function(c)
                if c:IsA("Humanoid") then conn:Disconnect(); task.wait(0.1); watchVictimAnimator(victimModel) end
            end)
            vScope:Add(conn); return
        end
        local va = vh:FindFirstChildOfClass("Animator")
        if not va then va = Instance.new("Animator"); va.Parent = vh end
        task.defer(function()
            for _, trk in ipairs(va:GetPlayingAnimationTracks()) do
                if trk.Animation then
                    local raw = tostring(trk.Animation.AnimationId or ""):match("%d+")
                    if raw and _activeExecPlayerIds[raw] then
                        local motorDefaults = {}
                        for _, d in ipairs(victimModel:GetDescendants()) do
                            if d:IsA("Motor6D") then motorDefaults[d] = {C0 = d.C0, C1 = d.C1} end
                        end
                        processVictimExecution(victimModel, va, trk, raw, motorDefaults, {})
                    end
                end
            end
        end)
        local motorDefaults = {}
        for _, d in ipairs(victimModel:GetDescendants()) do
            if d:IsA("Motor6D") then motorDefaults[d] = {C0 = d.C0, C1 = d.C1} end
        end
        local vRep = {}
        vScope:Connect(va.AnimationPlayed, function(trk)
            if not trk.Animation then return end
            local raw = tostring(trk.Animation.AnimationId or ""):match("%d+")
            if not (raw and _activeExecPlayerIds[raw]) then return end
            processVictimExecution(victimModel, va, trk, raw, motorDefaults, vRep)
        end)
    end

    local function findVictim()
        local now = tick()
        local best, bestT, bestKey = nil, 0, nil
        -- 👽 FIX: за один проход чистим протухшие/сломанные записи (TTL 0.3s + parent-check),
        -- иначе таблица растёт неограниченно — ключи-модели держатся живыми ссылкой и не GC-ятся.
        for vm, info in pairs(_pendingPlayerRig) do
            local alive = vm.Parent and info[2] and info[2].Parent
            local fresh = (now - info[1]) < 0.3
            if alive and fresh then
                if info[1] > bestT then bestT = info[1]; best = info[2]; bestKey = vm end
            else
                _pendingPlayerRig[vm] = nil
            end
        end
        if best then
            -- Использованную запись тоже убираем — повторное использование смысла не имеет.
            if bestKey then _pendingPlayerRig[bestKey] = nil end
            return best
        end
        if not (killerRoot and killerRoot.Parent) then return nil end
        local kpos = killerRoot.Position
        local closest, closestDist = nil, 8
        local function check(model)
            local h = model:FindFirstChildOfClass("Humanoid")
            local r = model:FindFirstChild("HumanoidRootPart") or model.PrimaryPart
            if h and r and h.Health > 0 and model ~= killerModel and model ~= jane then
                local d = (r.Position - kpos).Magnitude
                if d < closestDist then closestDist = d; closest = r end
            end
        end
        local pf = Workspace:FindFirstChild("Players")
        local sf = pf and pf:FindFirstChild("Survivors")
        if sf then for _, m in ipairs(sf:GetChildren()) do check(m) end end
        local mapF = Workspace:FindFirstChild("Map")
        local npcsF = mapF and mapF:FindFirstChild("Lobby") and mapF.Lobby:FindFirstChild("NPCs")
        if npcsF then for _, m in ipairs(npcsF:GetChildren()) do check(m) end end
        return closest
    end

    local function limpKiller()
        local rig = rigTarget
        if not rig then return nil end
        local saved = {}
        local torso = rig:FindFirstChild("Torso")
        if not torso then return saved end
        local function bend(name, angles)
            local m = torso:FindFirstChild(name)
            if m and m:IsA("Motor6D") then saved[m] = m.C0; m.C0 = m.C0 * CFrame.Angles(angles.X, angles.Y, angles.Z) end
        end
        bend("Neck", Vector3.new(math.rad(50), 0, 0))
        bend("Right Shoulder", Vector3.new(math.rad(80), 0, 0))
        bend("Left Shoulder", Vector3.new(math.rad(80), 0, 0))
        return saved
    end
    local function unlimpKiller(saved) for m, c0 in pairs(saved) do if m and m.Parent then m.C0 = c0 end end end

    local function tweenCFrame(part, fromCF, toCF, duration)
        local elapsed = 0
        while elapsed < duration do
            local dt = RunService.RenderStepped:Wait()
            elapsed = elapsed + dt
            if not (part and part.Parent) then return end
            local a = math.min(elapsed / duration, 1)
            local eased = a * a * (3 - 2 * a)
            part.CFrame = fromCF:Lerp(toCF, eased)
        end
    end

    local function playGhostExecution(execDuration)
        if _ghostExecBusy then return end
        if not ghostExecTrack then return end
        _ghostExecBusy = true
        execDuration = execDuration or EXEC_DURATION
        task.wait(0.05)
        local victimRoot = findVictim()
        if not victimRoot then _ghostExecBusy = false; return end
        if not (killerRoot and killerRoot.Parent and janeRoot and janeRoot.Parent) then _ghostExecBusy = false; return end
        local saved = limpKiller()
        for _, t in pairs(loaded.Normal) do if t.IsPlaying then t:Stop(0) end end
        for _, t in pairs(loaded.LMS) do if t.IsPlaying then t:Stop(0) end end
        local prevRender = renderConn
        if prevRender then prevRender:Disconnect() end
        local startCF = janeRoot.CFrame
        local targetCF = killerRoot.CFrame
        tweenCFrame(janeRoot, startCF, targetCF, 0.35)
        local waitedLoad = 0
        while ghostExecTrack.Length <= 0 and waitedLoad < 2 do task.wait(0.05); waitedLoad = waitedLoad + 0.05 end
        ghostExecTrack:Play(0.05, 1, 1)
        local sceneDuration = math.max(execDuration - 0.35 - waitedLoad - 0.05, 0.5)
        task.wait(sceneDuration)
        pcall(function() ghostExecTrack:Stop(0.1) end)
        if killerRoot and killerRoot.Parent and janeRoot and janeRoot.Parent then

    -- ===== applyLMSAnims: walking-skeleton анимации после Jane-трансформации =====
    -- Навешивает Idle/Move анимации на LP.Character (теперь это c00lkidd-skin после Jane'овской
    -- LMS-трансформации). Подвешивает heartbeat-цикл idle/move, footstep markers.
    local function applyLMSAnims_impl()
        local char = LP.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        if not hum then return end
        local animate = char:FindFirstChild("Animate")
        if animate then pcall(function() animate.Disabled = true end) end
        for _, t in pairs(hum:GetPlayingAnimationTracks()) do pcall(function() t:Stop(0) end) end
        local function loadTrack(id, prio)
            local a = Instance.new("Animation"); a.AnimationId = id
            local ok, t = pcall(function() return hum:LoadAnimation(a) end)
            if not (ok and t) then return nil end
            t.Priority, t.Looped = prio, true; return t
        end
        local idle = loadTrack(KILLER_LMS_ANIMS.Idle, Enum.AnimationPriority.Action)
        local move = loadTrack(KILLER_LMS_ANIMS.Move, Enum.AnimationPriority.Movement)
        if not (idle and move) then return end
        idle:Play(0.1); _killerAnimState = {idle = idle, move = move, animate = animate, hum = hum, char = char}
        local stepToggle = false
        local function playFootstep()
            if not _lmsSfxActive then return end
            local hrp = hum.RootPart or char:FindFirstChild("HumanoidRootPart")
            if not hrp then return end
            local id = stepToggle and _activeFootstep1 or _activeFootstep2; stepToggle = not stepToggle
            if id then pcall(function() local s = Instance.new("Sound"); s.SoundId, s.Volume, s.Parent = id, 1, hrp; s:Play(); Debris:AddItem(s, 2) end) end
        end
        for _, n in ipairs{"Footstep", "LeftFoot", "RightFoot", "Step"} do
            pcall(function() lmsTrack(move:GetMarkerReachedSignal(n):Connect(function() if _lmsSfxActive then playFootstep() end end)) end)
        end
        lmsTrack(move.KeyframeReached:Connect(function(kfName)
            if not _lmsSfxActive then return end
            local l = kfName and kfName:lower() or ""
            if l:find("step") or l:find("foot") then playFootstep() end
        end))
        lmsTrack(hum.AnimationPlayed:Connect(function(trk)
            if not _lmsSfxActive then return end
            local id = trk.Animation and trk.Animation.AnimationId
            if id == KILLER_LMS_ANIMS.Idle or id == KILLER_LMS_ANIMS.Move then return end
            if trk.Priority == Enum.AnimationPriority.Core then pcall(function() trk:Stop(0) end) end
        end))
        local hbConn
        hbConn = lmsTrack(RunService.Heartbeat:Connect(function()
            if not (hum and hum.Parent) then return end
            if hum.MoveDirection.Magnitude > 0.01 then
                if not move.IsPlaying then idle:Stop(0.1); move:Play(0.1) end
            else
                if not idle.IsPlaying then move:Stop(0.1); idle:Play(0.1) end
            end
        end))
        lmsTrack(hum.Died:Connect(function()
            if hbConn then hbConn:Disconnect() end
            if deactivateLMSSfx then pcall(deactivateLMSSfx) end
        end))
    end

-- rbxm-скина: накладываем ТОЛЬКО на VIP.
--
-- ПОЧЕМУ:
-- ReplicatedStorage.Assets.Skins.Killers.Slasher.VIPSlasher — это игровой ассет VIP-
-- варианта. Когда игрок экипирует его, движок собирает рig Slasher с body-частями
-- из VIPSlasher напрямую на верхнем уровне (Necrobloxian/Mask/Knife на уровне самого
-- km). Обычный Slasher же оборачивает свои части в подмодель "Slasher Export" внутри
-- рига. Этого достаточно для надёжного детекта без всяких RemoteEvent-проб.
--
-- ЧТО ДЕЛАЕТ:
-- 1. isVipRig(km) — true если в риге нет "Slasher Export" (= VIP-вариант).
-- 2. findVipMile1(km) — возвращает target-контейнер для applyParts:
--    - обычный Slasher: подмодель "Slasher Export"
--    - VIP Slasher: сам km (части лежат на верхнем уровне).
-- 3. shouldApplySkin(actorName, km) — главный гейт: для актёра Slasher разрешает
--    скин только если рig VIP. Для не-Slasher актёров — всегда true (логика их
--    скинов не меняется этим модулем).
--
-- ВЗАИМОДЕЙСТВИЕ С applyKillerSkin:
-- Раньше applyKillerSkin искал rigTarget через `Model :IsA + :FindFirstChildOfClass("MeshPart")`.
-- Эта проверка для VIP-рига НЕ срабатывает (нет model-обёртки с meshpart'ами) и функция
-- молча выходила по таймауту. Теперь applyKillerSkin для Slasher'а сначала зовёт
-- VIPSlasher.findVipMile1 и использует его результат как target.
--=========================================
do
    if not VIP_SLASHER_ONLY_ENABLED then
        zlog("CFG", "👑 VIP SLASHER module DISABLED (gate off, скин накладывается на любого Slasher)")
    else
        local EXPORT_NAME = "Slasher Export"  -- подмодель в риге обычного (не-VIP) Slasher'а

        -- isVipRig: true если у Slasher-рига нет подмодели "Slasher Export".
        -- Защита от мусора: проверяем что km вообще существует.
        local function isVipRig(km)
            if not (km and km.Parent) then return false end
            return km:FindFirstChild(EXPORT_NAME) == nil
        end

        -- findVipMile1: возвращает target-контейнер для applyParts.
        -- Для обычного Slasher = подмодель "Slasher Export" (там лежат body-части).
        -- Для VIP = сам km. Никогда не возвращает nil для валидного km.
        local function findVipMile1(km)
            if not (km and km.Parent) then return nil end
            local export = km:FindFirstChild(EXPORT_NAME)
            if export then return export end
            return km
        end

        -- shouldApplySkin: главный гейт. Для актёра Slasher — только VIP. Для остальных
        -- актёров (c00lkidd, 007n7, guest1337 и т.д.) — всегда true: их скины и логика
        -- этим модулем не затрагиваются.
        local function shouldApplySkin(actorName, km)
            if actorName ~= "Slasher" then return true end
            return isVipRig(km)
        end

        VIPSlasher.isVipRig        = isVipRig
        VIPSlasher.findVipMile1    = findVipMile1
        VIPSlasher.shouldApplySkin = shouldApplySkin

        zlog("CFG", "👑 VIP SLASHER module initialized (Slasher skin gated to VIP only)")
    end
end

    -- Допуск: если кадр пропустил точный момент (например лагнуло), щелчок всё равно
    -- сработает если TimePosition в пределах TOLERANCE после тайминга. Если перевалил
    -- больше чем на это — щелчок считается пропущенным, переходим к следующему.
    -- 0.15s = достаточно чтобы пережить заметный фриз, но не настолько много чтобы
    -- "догонять" каждый пропущенный щелчок в позднюю часть трека.
    local CLICK_TOLERANCE = 0.15

    local FLASH_BOOST     = 0.15     -- альфа золотой вспышки (0..1) на пике
    local FLASH_COLOR     = Color3.fromRGB(255, 200, 60)   -- золотой
    local CHROM_BOOST     = 0.1     -- величина RGB-сдвига через TintColor (макс 1.0)
    local DECAY_RATE      = 8.0      -- exp(-dt*8) → 50% за ~87ms
    local SHAKE_DECAY     = 15.0     -- shake затухает БЫСТРЕЕ чем FOV/roll (~50% за 46ms) —
                                     -- импульсная тряска должна быть резкой
    local THRESHOLD_MULT  = 1.35     -- удар = current > baseline * 1.35
    -- дополнительное требование чтобы отсечь сустейны (вокал/синт растут плавно,
    -- удар = резкий скачок за 1 кадр). Сустейн может превысить baseline*1.35 если растёт
    -- медленно, и старого условия (loudness > prev) недостаточно — рост на 0.5/кадр
    -- тоже > prev, но это не удар. RISE_MIN требует рост МИНИМУМ на это значение за
    -- 1 кадр — транзиент даёт +20…+60, сустейн +1…+5.
    local RISE_MIN        = 20       -- мин (loudness - prev) для подтверждения удара
    -- refractory увеличен с 0.10 → 0.25. Раньше один kick (длится 30-80ms
    -- и содержит несколько внутренних пиков: атака+саб+хвост) триггерил детектор
    -- 2-3 раза подряд. 0.25s гарантирует что внутри одного удара триггер сработает
    -- ровно один раз. Минус — на быстрой музыке (>240 BPM, dnb-стиль) пропустит
    -- 16-е ноты; если такое встретится, снизить до 0.15-0.18.
    local REFRACTORY      = 0.25
    local WINDOW_SIZE     = 30       -- baseline window ~0.5s @ 60fps
    local MIN_LOUDNESS    = 5
    local RENDER_BIND_NAME = "ZORG_SLASHER_BEAT_VISUAL"
    local COLOR_FX_NAME   = "ZorgSlasherBeatColorFX"
    local FLASH_GUI_NAME  = "ZorgSlasherBeatFlash"

    local _flashDelta, _chromDelta = 0, 0        -- импульсы post-effects
    local _rollSignToggle = 1
    local _colorFX = nil                         -- ColorCorrectionEffect (chromatic) в Lighting

    -- Возвращает "layer1"/"layer2"/"layer3"/"layer4" | "lms" | nil. Смотрим
    -- origIds[snd] (исходный SoundId до подмены через infectSound), потому что после
    -- подмены snd.SoundId — это уже наш custom asset, который не в _activeChaseMap.
    -- VISUAL_LAYER_MAP содержит только VIP-Slasher ID. Для других скинов вернём nil
    -- — визуал работает ИСКЛЮЧИТЕЛЬНО для VIP, для остальных тихо игнорим.
    -- LMS-звуки (LMS_ID) — отдельная ветка с таблицами таймингов.
    local function classifySound(snd)
        if not (snd and snd.Parent) then return nil end
        if snd.Name:match("^ZORG_PHANTOM") then return nil end
        local raw = tostring(origIds[snd] or snd.SoundId or ""):match("%d+")
        if not raw then return nil end
        if raw == LMS_ID then return "lms" end
        local layer = VISUAL_LAYER_MAP[raw]
        if layer then return layer end
        return nil
    end

    local function detectBeat(state, loudness, now)
        if now - state.lastBeat < REFRACTORY then
            table.insert(state.window, loudness)
            if #state.window > WINDOW_SIZE then table.remove(state.window, 1) end
            state.prev = loudness
            return false
        end
        local sum = 0
        for _, v in ipairs(state.window) do sum = sum + v end
        local baseline = #state.window > 0 and (sum / #state.window) or loudness
        table.insert(state.window, loudness)
        if #state.window > WINDOW_SIZE then table.remove(state.window, 1) end
        local isBeat = loudness > MIN_LOUDNESS
            and loudness > baseline * THRESHOLD_MULT
            and loudness > state.prev
            and (loudness - state.prev) > RISE_MIN   -- транзиент, не сустейн
        if isBeat then state.lastBeat = now end
        state.prev = loudness
        return isBeat
    end

    -- Аддитивные импульсы с cap'ом, чтобы серия ударов не зашкаливала.
    -- kind: "layer1".."layer4" — VIP-Slasher по taxonomy LAYER_MULTS (см. выше)
    --       "lms"              — FOV+shake+flash+chroma на полную; roll отдельно через onClick
    --       nil/иное           — обрабатываем как layer4 (полный профиль) — для testBeat
    local function onBeat(kind)
        -- LMS — всегда на полную, кроме roll (он на щелчках через onClick)
        if kind == "lms" then
            _fovDelta = math.min(_fovDelta + FOV_BOOST, FOV_BOOST * 1.6)
            _shakeDelta = math.min(_shakeDelta + SHAKE_BOOST, SHAKE_BOOST * 1.6)
            _flashDelta = math.min(_flashDelta + FLASH_BOOST, FLASH_BOOST * 1.6)
            _chromDelta = math.min(_chromDelta + CHROM_BOOST, CHROM_BOOST * 1.6)
            return
        end

        -- Подбираем множители; если kind неизвестен (например testBeat без kind), даём полный.
        local m = LAYER_MULTS[kind] or LAYER_MULTS.layer4

        if m.fov > 0 then
            local b = FOV_BOOST * m.fov
            _fovDelta = math.min(_fovDelta + b, b * 1.6)
        end
        if m.shake > 0 then
            local b = SHAKE_BOOST * m.shake
            _shakeDelta = math.min(_shakeDelta + b, b * 1.6)
        end
        if m.flash > 0 then
            local b = FLASH_BOOST * m.flash
            _flashDelta = math.min(_flashDelta + b, b * 1.6)
        end
        if m.chrom > 0 then
            local b = CHROM_BOOST * m.chrom
            _chromDelta = math.min(_chromDelta + b, b * 1.6)
        end
        if m.roll > 0 then
            local rollMag = ROLL_BOOST * m.roll
            local newRoll = _rollDelta + rollMag * _rollSignToggle
            local maxAbs = rollMag * 1.6
            if math.abs(newRoll) > maxAbs then newRoll = (newRoll > 0) and maxAbs or -maxAbs end
            _rollDelta = newRoll
            _rollSignToggle = -_rollSignToggle
        end
    end

    -- Импульс roll'а от щелчка. Слабее основного (LMS_ROLL_MULT = 0.5 → ±4°).
    -- Вызывается из cameraStep по таблице LMS_CLICK_TIMINGS.
    local function onClick()
        local rollMag = ROLL_BOOST * LMS_ROLL_MULT
        local newRoll = _rollDelta + rollMag * _rollSignToggle
        local maxAbs = rollMag * 1.6
        if math.abs(newRoll) > maxAbs then newRoll = (newRoll > 0) and maxAbs or -maxAbs end
        _rollDelta = newRoll
        _rollSignToggle = -_rollSignToggle
    end

    local function attachSound(snd)
        if _watchedSounds[snd] then return end
        local kind = classifySound(snd)
        if not kind then return end
        local state = {window = {}, prev = 0, lastBeat = 0, kind = kind}
        -- Для LMS добавляем указатель на следующий щелчок в таблице таймингов.
        -- Стартуем с первого тайминга, который ещё не прошёл (на случай attach
        -- посреди трека — например, если скрипт стартанул после начала музыки).
        if kind == "lms" and #LMS_CLICK_TIMINGS > 0 then
            local pos = snd.TimePosition or 0
            state.nextClickIndex = 1
            for i, t in ipairs(LMS_CLICK_TIMINGS) do
                if t > pos then state.nextClickIndex = i; break end
                state.nextClickIndex = i + 1   -- все тайминги уже прошли
            end
        end
        -- аналогично для битов. Если таблица непустая — автодетектор
        -- для этого LMS-звука будет выключен (см. cameraStep).
        if kind == "lms" and #LMS_BEAT_TIMINGS > 0 then
            local pos = snd.TimePosition or 0
            state.nextBeatIndex = 1
            for i, t in ipairs(LMS_BEAT_TIMINGS) do
                if t > pos then state.nextBeatIndex = i; break end
                state.nextBeatIndex = i + 1
            end
        end
        -- VIP-Slasher: индивидуальный массив таймингов на каждый layer.
        -- В отличие от LMS — автодетектор для VIP отключён всегда (см. cameraStep),
        -- поэтому если массив пустой, на этом layer'е просто не будет битов.
        local layerTable = LAYER_BEAT_TIMINGS_MAP[kind]
        if layerTable and #layerTable > 0 then
            local pos = snd.TimePosition or 0
            state.nextBeatIndex = 1
            for i, t in ipairs(layerTable) do
                if t > pos then state.nextBeatIndex = i; break end
                state.nextBeatIndex = i + 1
            end
        end
        _watchedSounds[snd] = state
        -- 👽 V7.92 SCOPE PHASE 3 (FIX 6 из Scope.lua): было table.insert(_watcherConns,
        -- snd.Destroying:Connect(...)) — на каждый звук создавался коннект, который висел
        -- в _watcherConns ДО конца stop(). На LMS+chase+voicelines = сотни мёртвых записей
        -- за матч. Сейчас :Once сам себя Disconnect'ит после срабатывания и удаляется из
        -- scope автоматически. Никакого накопления.
        _beatScope:Once(snd.Destroying, function()
            _watchedSounds[snd] = nil
        end)
    end

    local function scanExisting()
        if not getinstances then return end
        for _, o in pairs(getinstances()) do
            if typeof(o) == "Instance" and o:IsA("Sound") then attachSound(o) end
        end
    end

    -- 👽 V7.91: общий хелпер для трёх идентичных паттернов trigger-по-таблице в cameraStep.
    -- Все три блока (clicks LMS / beats LMS / beats VIP-layer) делали одно и то же:
    -- 1) detect перемотки через pos < prevT - 0.5 → пересчёт индекса,
    -- 2) while nextIndex <= #timings: if pos >= target → trigger, advance index.
    -- Различались только: indexKey, массив таймингов, tolerance, onTrigger.
    -- Возвращает новый индекс (state[indexKey] уже обновлён внутри).
    local function tickTimingTable(state, indexKey, timings, tolerance, pos, onTrigger)
        local idx = state[indexKey]
        if not idx then return end
        -- Detect перемотки: если позиция меньше предыдущего сработавшего тайминга −0.5s
        -- (запас на люфт TimePosition) — перебираем массив заново с начала.
        if idx > 1 then
            local prevT = timings[idx - 1]
            if prevT and pos < prevT - 0.5 then
                idx = 1
                for i, t in ipairs(timings) do
                    if t > pos then idx = i; break end
                    idx = i + 1
                end
            end
        end
        -- Trigger: пробегаем все тайминги, которые попали в окно [pos-tolerance, pos].
        -- Один кадр может покрыть несколько таймингов (плотные удары / freeze-кадр).
        while idx <= #timings do
            local target = timings[idx]
            if pos < target then break end
            if pos - target <= tolerance then
                onTrigger()
            end
            idx = idx + 1
        end
        state[indexKey] = idx
    end

    -- Beat-detection + click-timing trigger + beat-timing trigger + decay + apply.
    local function cameraStep(dt)
        local now = tick()
        for snd, state in pairs(_watchedSounds) do
            if snd.Parent and snd.IsPlaying then
                -- Автодетектор отключается:
                --   1) для LMS, если задана таблица битов;
                --   2) для VIP-layer'ов всегда (kind в layer1..layer4) — VIP работает
                --      ТОЛЬКО по таблицам, потому что PlaybackLoudness не справлялся
                --      ни на одном из 4 треков. Если таблица пустая — на этом layer'е
                --      просто не будет битов, и это норма (Layer 1 может быть пуст).
                local isVipLayer = LAYER_BEAT_TIMINGS_MAP[state.kind] ~= nil
                local useAutoDetect = not isVipLayer
                    and not (state.kind == "lms" and state.nextBeatIndex)
                if useAutoDetect then
                    local ok, loudness = pcall(function() return snd.PlaybackLoudness end)
                    if ok and loudness then
                        if detectBeat(state, loudness, now) then onBeat(state.kind) end
                    end
                end

                -- 👽 V7.91: три идентичных паттерна заменены на вызовы tickTimingTable.
                -- TimePosition читается один раз и переиспользуется для обоих LMS-блоков.
                if state.kind == "lms" and (state.nextClickIndex or state.nextBeatIndex) then
                    local okPos, pos = pcall(function() return snd.TimePosition end)
                    if okPos and pos then
                        -- триггер щелчков (LMS_CLICK_TIMINGS → onClick → roll)
                        if state.nextClickIndex then
                            tickTimingTable(state, "nextClickIndex", LMS_CLICK_TIMINGS, CLICK_TOLERANCE, pos, onClick)
                        end
                        -- триггер битов (LMS_BEAT_TIMINGS → onBeat("lms") → FOV/shake/flash, без roll)
                        if state.nextBeatIndex then
                            tickTimingTable(state, "nextBeatIndex", LMS_BEAT_TIMINGS, CLICK_TOLERANCE, pos, function() onBeat("lms") end)
                        end
                    end
                end

                -- триггер битов для VIP-layer'ов (индивидуальные таблицы из LAYER_BEAT_TIMINGS_MAP).
                -- VIP-треки лупятся → защита от перемотки внутри tickTimingTable автоматически
                -- обрабатывает начало нового цикла (pos < prevT-0.5 → индекс сбрасывается).
                if isVipLayer and state.nextBeatIndex then
                    local layerT = LAYER_BEAT_TIMINGS_MAP[state.kind]
                    if layerT then
                        local okPos, pos = pcall(function() return snd.TimePosition end)
                        if okPos and pos then
                            local sk = state.kind
                            tickTimingTable(state, "nextBeatIndex", layerT, CLICK_TOLERANCE, pos, function() onBeat(sk) end)
                        end
                    end
                end
            elseif not snd.Parent then
                _watchedSounds[snd] = nil
            end
        end

        -- Decay все импульсы
        local k     = math.exp(-dt * DECAY_RATE)
        local kSh   = math.exp(-dt * SHAKE_DECAY)
        _fovDelta, _rollDelta = _fovDelta * k, _rollDelta * k
        _shakeDelta = _shakeDelta * kSh
        _flashDelta, _chromDelta = _flashDelta * k, _chromDelta * k

        local cam = Workspace.CurrentCamera
        if cam then
            if _fovDelta > 0.01 then cam.FieldOfView = cam.FieldOfView + _fovDelta end
            -- Position shake вперёд по view-вектору. Применяем ПЕРЕД roll, чтобы roll
            -- не вращался вокруг сдвинутой точки.
            if _shakeDelta > 0.01 then
                cam.CFrame = cam.CFrame + cam.CFrame.LookVector * _shakeDelta
            end
            if math.abs(_rollDelta) > 0.001 then
                cam.CFrame = cam.CFrame * CFrame.Angles(0, 0, math.rad(_rollDelta))
            end
        end

        -- Золотая вспышка через полупрозрачный Frame в ScreenGui.
        if _flashEdges then
            local trans = 1 - math.min(_flashDelta, 1)
            for _, edge in ipairs(_flashEdges) do
                if edge.Parent then
                    edge.BackgroundTransparency = trans
                end
            end
        end

        -- Псевдо-chromatic aberration через ColorCorrectionEffect.TintColor.
        -- Сдвиг в красное при положительном roll, в синее при отрицательном — цветовой
        -- сдвиг следует за направлением наклона. Не настоящий RGB-split (нужен шейдер),
        -- но даёт похожее ощущение распада картинки на пиках.
        if _colorFX and _colorFX.Parent then
            if _chromDelta > 0.005 then
                local sign = (_rollDelta >= 0) and 1 or -1
                local shift = _chromDelta * sign
                _colorFX.TintColor = Color3.new(
                    1 + math.max(shift, 0)  * 0.5,
                    1 - math.abs(shift)     * 0.3,
                    1 + math.max(-shift, 0) * 0.5
                )
            else
                _colorFX.TintColor = Color3.new(1, 1, 1)
            end
        end
    end

    -- ===== PUBLIC API =====
    function SlasherBeatVisual.start()
        if _running then return end
        _running = true
        _fovDelta, _rollDelta, _rollSignToggle = 0, 0, 1
        _shakeDelta = 0
        _flashDelta, _chromDelta = 0, 0
        -- ColorCorrectionEffect для chromatic-aberration (TintColor сдвиги).
        if not _colorFX or not _colorFX.Parent then
            _colorFX = Instance.new("ColorCorrectionEffect")
            _colorFX.Name = COLOR_FX_NAME
            _colorFX.Brightness = 0
            _colorFX.TintColor = Color3.new(1, 1, 1)
            _colorFX.Parent = Lighting
        end
        -- ScreenGui + 4 Frame'а по краям экрана (top/bottom/left/right) с UIGradient
        -- для плавного затухания от края к центру. Не требует внешних ассетов.
        -- В покое все 4 Frame'а имеют BackgroundTransparency=1 (невидимы). На пике
        -- бита cameraStep ставит им 1-alpha синхронно. UIGradient.Transparency
        -- домножается поверх — получается виньетка с плавной альфой.
        -- THICKNESS — толщина свечения от края к центру (доля экрана).
        if not _flashGui or not _flashGui.Parent then
            local PlayerGui = LP:FindFirstChildOfClass("PlayerGui") or LP:WaitForChild("PlayerGui")
            _flashGui = Instance.new("ScreenGui")
            _flashGui.Name = FLASH_GUI_NAME
            _flashGui.IgnoreGuiInset = true
            _flashGui.ResetOnSpawn = false
            _flashGui.DisplayOrder = 9999
            _flashGui.Parent = PlayerGui

            local THICKNESS = 0.25  -- 25% экрана от каждого края
            _flashEdges = {}   -- список 4 Frame'ов для одновременного апдейта

            local function makeEdge(name, anchor, pos, size, gradientRotation)
                local f = Instance.new("Frame")
                f.Name = name
                f.AnchorPoint = anchor
                f.Position = pos
                f.Size = size
                f.BackgroundColor3 = FLASH_COLOR
                f.BackgroundTransparency = 1   -- невидимо в покое
                f.BorderSizePixel = 0
                f.Active = false
                f.ZIndex = 100
                f.Parent = _flashGui
                local g = Instance.new("UIGradient")
                -- От плотного (внешний край) к прозрачному (к центру).
                g.Transparency = NumberSequence.new({
                    NumberSequenceKeypoint.new(0, 0),    -- внешний край: непрозрачно
                    NumberSequenceKeypoint.new(1, 1),    -- внутренний край: прозрачно
                })
                g.Rotation = gradientRotation
                g.Parent = f
                table.insert(_flashEdges, f)
                return f
            end

            -- TOP: gradient идёт сверху вниз → rotation=90 даёт alpha от 0 (наверху) к 1 (внизу)
            makeEdge("Top",    Vector2.new(0, 0),     UDim2.fromScale(0, 0),
                     UDim2.fromScale(1, THICKNESS),   90)
            -- BOTTOM: alpha 0 внизу, 1 наверху → rotation=270 (или -90)
            makeEdge("Bottom", Vector2.new(0, 1),     UDim2.fromScale(0, 1),
                     UDim2.fromScale(1, THICKNESS),   270)
            -- LEFT: alpha 0 слева, 1 справа → rotation=0
            makeEdge("Left",   Vector2.new(0, 0),     UDim2.fromScale(0, 0),
                     UDim2.fromScale(THICKNESS, 1),   0)
            -- RIGHT: alpha 0 справа, 1 слева → rotation=180
            makeEdge("Right",  Vector2.new(1, 0),     UDim2.fromScale(1, 0),
                     UDim2.fromScale(THICKNESS, 1),   180)

            -- _flashEdges[1].Parent служит проверкой живости виньетки в cameraStep.
            -- Если GUI убит снаружи — _flashEdges[1].Parent = nil, и блок flash пропускается.
        end
        scanExisting()
        for _, svc in ipairs({Workspace, game:GetService("SoundService"), LP, RS, game:GetService("CoreGui")}) do
            -- 👽 V7.92 SCOPE PHASE 3: было table.insert(_watcherConns, svc.DescendantAdded:Connect(...))
            _beatScope:Connect(svc.DescendantAdded, function(d)
                if _running and d:IsA("Sound") then attachSound(d) end
            end)
        end
        -- 👽 V7.92 SCOPE PHASE 5+: Scope:BindToRenderStep — RunService:Bind + регистрация
        -- cleanup-функции (UnbindFromRenderStep) в один вызов. Раньше: ручной Bind + флаг

    -- ===== ТЕСТОВЫЕ КНОПКИ =====
    -- Helper: запускает temporary bind на N секунд со своим temp ScreenGui+ColorFX
    -- (чтобы не конфликтовать с running-режимом). На каждом кадре зовёт beatScheduler(t)
    -- — callback, который сам решает когда ставить onBeat и какого kind (даём ему
    -- elapsed time, он смотрит и возвращает kind или nil). Это позволяет одной функцией
    -- покрыть и одиночный chase-beat, и LMS-последовательность с переходом drop.
    local function _runTestSequence(durationSec, beatScheduler)
        local TEST_BIND = RENDER_BIND_NAME .. "_TEST"
        local TEST_FX   = COLOR_FX_NAME .. "_TEST"
        local TEST_GUI  = FLASH_GUI_NAME .. "_TEST"
        pcall(function() RunService:UnbindFromRenderStep(TEST_BIND) end)
        local oldFX = Lighting:FindFirstChild(TEST_FX)
        if oldFX then pcall(function() oldFX:Destroy() end) end
        local PlayerGui = LP:FindFirstChildOfClass("PlayerGui") or LP:WaitForChild("PlayerGui")
        local oldGui = PlayerGui:FindFirstChild(TEST_GUI)
        if oldGui then pcall(function() oldGui:Destroy() end) end

        local fx = Instance.new("ColorCorrectionEffect")
        fx.Name = TEST_FX
        fx.TintColor = Color3.new(1, 1, 1)
        fx.Parent = Lighting
        local gui = Instance.new("ScreenGui")
        gui.Name = TEST_GUI
        gui.IgnoreGuiInset = true
        gui.DisplayOrder = 9999
        gui.Parent = PlayerGui
        local frame = Instance.new("Frame")
        frame.Size = UDim2.fromScale(1, 1)
        frame.BackgroundColor3 = FLASH_COLOR
        frame.BackgroundTransparency = 1
        frame.BorderSizePixel = 0
        frame.ZIndex = 100
        frame.Active = false
        frame.Parent = gui

        _fovDelta, _rollDelta, _flashDelta, _chromDelta = 0, 0, 0, 0
        local elapsed = 0
        RunService:BindToRenderStep(TEST_BIND, Enum.RenderPriority.Camera.Value + 1, function(dt)
            elapsed = elapsed + dt
            -- Сначала спрашиваем у scheduler'а — пора ли ставить beat?
            local kindToBeat = beatScheduler(elapsed, dt)
            if kindToBeat then onBeat(kindToBeat) end

            -- Decay + apply (как в основном cameraStep, но в свой fx/gui)
            local k   = math.exp(-dt * DECAY_RATE)
            local kSh = math.exp(-dt * SHAKE_DECAY)
            _fovDelta, _rollDelta = _fovDelta * k, _rollDelta * k
            _shakeDelta = _shakeDelta * kSh
            _flashDelta, _chromDelta = _flashDelta * k, _chromDelta * k
            local c = Workspace.CurrentCamera
            if c then
                if _fovDelta > 0.01 then c.FieldOfView = c.FieldOfView + _fovDelta end
                if _shakeDelta > 0.01 then
                    c.CFrame = c.CFrame + c.CFrame.LookVector * _shakeDelta
                end
                if math.abs(_rollDelta) > 0.001 then
                    c.CFrame = c.CFrame * CFrame.Angles(0, 0, math.rad(_rollDelta))
                end
            end
            if frame.Parent then
                frame.BackgroundTransparency = 1 - math.min(_flashDelta, 1)
            end
            if fx.Parent then
                if _chromDelta > 0.005 then
                    local sign = (_rollDelta >= 0) and 1 or -1
                    local shift = _chromDelta * sign
                    fx.TintColor = Color3.new(
                        1 + math.max(shift, 0)  * 0.5,
                        1 - math.abs(shift)     * 0.3,
                        1 + math.max(-shift, 0) * 0.5
                    )
                else
                    fx.TintColor = Color3.new(1, 1, 1)
                end
            end
            if elapsed > durationSec then
                pcall(function() RunService:UnbindFromRenderStep(TEST_BIND) end)
                _fovDelta, _rollDelta, _flashDelta, _chromDelta = 0, 0, 0, 0
                _shakeDelta = 0
                pcall(function() fx:Destroy() end)
                pcall(function() gui:Destroy() end)
            end
        end)
    end

            local candidate = VIPSlasher.findVipMile1(km)

                rigTarget = candidate; break

            for _, c in ipairs(km:GetChildren()) do if c:IsA("Model") and c:FindFirstChildOfClass("MeshPart") then rigTarget = c; break end end

    task.spawn(C00lkidd.onKillerSpawn, actorInfo, km, rigTarget)

-- 👽 V7.69: общий helper для двух очень похожих attach*KfsHandler-функций.
-- Раньше attachKfsExecHandler и attachVictimKfsHandler дублировали ~50 строк wait-логики
-- (ждать Humanoid → ждать Animator → проверить marker → запустить onReady). Различались
-- только именем атрибута-маркера и телом onReady. Теперь wait-логика общая, в неё
-- передаётся (markerAttr, onReady). Каждая функция содержит ТОЛЬКО свою уникальную часть
-- (preload + AnimationPlayed listener).
--
-- Логика wait-цепочки идентична оригиналу (зафиксирована в V7.63 LEAK FIXES для killer-side):
--   1. Проверяем marker — если уже установлен, выходим (защита от двойной установки)
--   2. Ищем Humanoid: есть → переходим к шагу 4; нет → подписываемся на ChildAdded
--   3. (callback ChildAdded) когда появился Humanoid, дисконнектимся и идём в шаг 4
--   4. Ищем Animator внутри Humanoid'а: есть → шаг 6; нет → подписываемся на Hum.ChildAdded
--   5. (callback Hum.ChildAdded) когда появился Animator, дисконнектимся и идём в шаг 6
--   6. setupListener — проверяем marker ещё раз (защита от race), ставим marker, вызываем onReady(hum, anim)
--
-- ВНИМАНИЕ: маркер проверяется в трёх местах (вход + два callback'а ChildAdded) — это
-- защита от того что Humanoid и Animator могут появиться "одновременно" (в одном кадре),
-- и оба пути setup'а сработают. Marker отсекает второй.
local function waitForHumanoidAnimator(model, markerAttr, onReady)
    if not model or model:GetAttribute(markerAttr) then return end

    local function trySetup(hum, anim)
        if model:GetAttribute(markerAttr) then return end
        model:SetAttribute(markerAttr, true)
        onReady(hum, anim)
    end

    local function waitForAnimator(hum)
        local anim = hum:FindFirstChildOfClass("Animator")
        if anim then trySetup(hum, anim); return end
        local conn
        conn = hum.ChildAdded:Connect(function(c)
            if c:IsA("Animator") then
                conn:Disconnect()
                trySetup(hum, c)
            end
        end)
        track(conn)
    end

    local hum = model:FindFirstChildOfClass("Humanoid")
    if hum then waitForAnimator(hum); return end
    local conn
    conn = model.ChildAdded:Connect(function(c)
        if c:IsA("Humanoid") then
            conn:Disconnect()
            waitForAnimator(c)
        end
    end)
    track(conn)
end

local function attachKfsExecHandler(actorInfo, killerModel)
    -- Хендлер навешивается, если есть хоть один KFS-вариант. Решение по конкретной
    -- анимации (брать KFS или дать игре fallback) принимается per-animation ниже.
    if not hasAnyKfsFile(actorInfo.ActorName) then return end

    waitForHumanoidAnimator(killerModel, "ZORG_KFS_ATTACHED", function(killerHum, kAnim)
        -- 👽 V7.64: generic preload — собираем уникальные variant'ы из _activeExecKillerIds
        -- (заполнен в loadKillerConfig) и прелоадим только те, для которых реально есть файл.

                preloaded[v] = true
                task.spawn(loadExecKFS, actorInfo.ActorName, v)
            end
        end

        -- 👽 V7.92 SCOPE PHASE 4: kAnim.AnimationPlayed теперь в currentKillerScope. Когда
        -- актёр уходит — scope:Destroy() убивает этот коннект сразу, а не оставляет dead-в-

        local execScope = currentKillerScope or _zorgScope
        execScope:Connect(kAnim.AnimationPlayed, function(trk)
            if not trk.Animation then return end
            local raw = tostring(trk.Animation.AnimationId or ""):match("%d+")
            local info = raw and _activeExecKillerIds[raw]
            if not info then return end
            -- Если для этого варианта нет KFS-файла, отдаём игре её обычную анимацию
            -- (она уже подменена на EXEC_ANIM_FALLBACK_ID в loadKillerConfig).
            if not hasKfsFile(actorInfo.ActorName, info.variant) then return end
            pcall(function() trk:Stop(0) end); pcall(function() trk:Destroy() end)
            playCustomExecAnim(actorInfo.ActorName, killerModel, killerHum, info.duration, info.variant)
        end)
    end)
end

-- 👽 V7.65: attachVictimKfsHandler — global victim-watcher для не-c00lkidd киллеров.
-- C00lkidd использует свою отдельную систему (клон жертвы внутри c00lkidd-модуля).
-- Для Slasher'а и других — этот watcher слушает Animator жертвы напрямую и при
-- срабатывании player-rig execution-анимации запускает playCustomVictimAnim, который
-- использует тот же KFS-плеер, что и для киллера, но пишет в motor.Transform жертвы.
--
-- Маркер ZORG_VICTIM_KFS_ATTACHED на victimModel — защита от двойной установки
-- (так же как ZORG_KFS_ATTACHED для killer-side).
local function attachVictimKfsHandler(victimModel)
    if not victimModel then return end

    waitForHumanoidAnimator(victimModel, "ZORG_VICTIM_KFS_ATTACHED", function(victimHum, vAnim)
        track(vAnim.AnimationPlayed:Connect(function(trk)
            if not trk.Animation then return end
            -- Текущий киллер? Если c00lkidd — отдаём c00lkidd-модулю (он использует клон).
            local actorInfo = _currentKiller
            if not actorInfo or not actorInfo.ActorName then return end
            if C00lkidd.isMyActor(actorInfo.ActorName) then return end

            local raw = tostring(trk.Animation.AnimationId or ""):match("%d+")
            local info = raw and _activeExecPlayerIds[raw]
            if not info then return end

            -- Проверяем что для этого variant есть victim KFS-файл — иначе пусть играет
            -- стандартная анимация (135641783638394 через __newindex подмену).
            if not hasVictimKfsFile(actorInfo.ActorName, info.variant) then return end

            -- Стопаем оригинальный track, запускаем наш KFS-плеер на rig жертвы.
            pcall(function() trk:Stop(0) end)
            pcall(function() trk:Destroy() end)
            playCustomVictimAnim(actorInfo.ActorName, victimModel, victimHum, info.duration, info.variant)
        end))
    end)
end

local function onKillerActorAdded(actorInfo)
    if not skinsEnabled then return end
    _currentKiller = actorInfo
    -- 👽 V7.92 SCOPE PHASE 4: пересоздаём currentKillerScope. Если предыдущий актёр ещё
    -- не был обработан onRemoved (например, быстрая смена) — его scope уничтожается здесь.
    -- Это запустит cleanup-функцию restoreRigTransparency для уходящего киллера.
    if currentKillerScope then
        currentKillerScope:Destroy()
    end
    currentKillerScope = _zorgScope:Child("killer_" .. actorInfo.ActorName)
    currentKillerScope.skinInfo = {}
    currentKillerScope.actorInfo = actorInfo
    -- Cleanup-функция: при scope:Destroy восстанавливаем Transparency на rigTarget-частях
    -- всех моделей этого киллера. Раньше это делалось вручную в onKillerActorRemoved
    -- (см. строки 4138-4146 v51) — теперь автоматически через scope-cleanup. snapshot
    -- захватывает текущую ссылку — на случай, если глобальная currentKillerScope уже
    -- переписана новым актёром к моменту выполнения cleanup (защита от теоретической гонки).
    do
        local snapshot = currentKillerScope
        snapshot:Cleanup(function()
            for _, info in pairs(snapshot.skinInfo) do
                if info and info.rigTarget then pcall(restoreRigTransparency, info.rigTarget) end
            end
        end)
    end
    resetLMSState(); loadKillerConfig(actorInfo)
    -- После смены конфига перейнфектим существующие звуки под нового киллера.
    -- clearActiveSoundMappings уже вернул их к оригиналам, теперь применяем новые мапинги.
    if masterEnabled and getinstances then
        for _, o in pairs(getinstances()) do
            if typeof(o) == "Instance" and o:IsA("Sound") then
                infectSound(o)
            end
        end
    end
    local kf = getKillersFolder()
    if not kf then return end
    local function tryApply()
        local km = actorInfo.Rig
        if not km then
            for _, c in ipairs(kf:GetChildren()) do
                if c.Name == actorInfo.Player.Name or c.Name:find(actorInfo.ActorName) then km = c; break end
            end
        end
        if km then
            task.spawn(applyKillerSkin, actorInfo, km)
            task.spawn(attachKfsExecHandler, actorInfo, km)
        end
    end
    tryApply()

    -- 🔪 Slasher-only: запуск beat-visual модуля. Для других актёров (включая c00lkidd)
    -- isMyActor вернёт false и start() — no-op. Безопасно звать всегда.
    -- 👑 V7.89: для Slasher beat-visual запускается ТОЛЬКО если рig — VIP. Без VIP
    -- наши кастомные layer-звуки всё равно не играют (их ID привязаны к VIPSlasher-
    -- ассету), так что и beat-detect был бы вхолостую.
    if SlasherBeatVisual.isMyActor(actorInfo.ActorName) then
        task.spawn(function()
            -- Рig может не успеть собраться к моменту ActorAdded — ждём появления.
            local km, waited = nil, 0
            repeat
                km = actorInfo.Rig
                if not km then
                    for _, c in ipairs(kf:GetChildren()) do
                        if c.Name == (actorInfo.Player and actorInfo.Player.Name) or c.Name:find(actorInfo.ActorName) then km = c; break end
                    end
                end
                if km then break end
                task.wait(0.1); waited = waited + 0.1
            until waited > 5
            -- Доп. ожидание появления признака VIP/обычный (Necrobloxian или Slasher Export).
            -- Если km нашли, но структура ещё пуста — дёргаем findVipMile1 несколько раз.
            local isVip = false
            if km then
                waited = 0
                repeat
                    if km:FindFirstChild("Slasher Export") or km:FindFirstChild("Necrobloxian") then
                        isVip = VIPSlasher.isVipRig(km)
                        break
                    end
                    task.wait(0.1); waited = waited + 0.1
                until waited > 5
            end
            if isVip then
                SlasherBeatVisual.start()
            else
                zlog("CFG", "👑 Slasher без VIP — beat-visual не запускается")
                SlasherBeatVisual.stop()
            end
        end)
    else
        -- Если предыдущий киллер был Slasher и сейчас сменился на другого — глушим.
        task.spawn(SlasherBeatVisual.stop)
    end
end

local function onKillerActorRemoved(actorInfo)
    if _currentKiller == actorInfo then _currentKiller = nil end
    if deactivateLMSSfx then pcall(deactivateLMSSfx) end
    _lmsTriggeredThisRound = false

    -- 👽 V7.92 SCOPE PHASE 4: ручная чистка _killerSkinInfo + restoreRigTransparency
    -- (раньше 10 строк on this place) свёрнута до scope:Destroy. Cleanup-функция,
    -- зарегистрированная в onKillerActorAdded, восстановит Transparency на rigTarget
    -- автоматически. Сам scope умрёт целиком (с потенциальным kAnim.AnimationPlayed
    -- коннектом и любыми future per-killer ресурсами).
    if currentKillerScope then
        currentKillerScope:Destroy()
        currentKillerScope = nil
    end

    -- 👽 FIX: GC orphan-записей _animReplacedCaches (Animator'ы у уничтоженных Humanoid'ов).
    -- deactivateLMSSfx чистит содержимое (треки), но ключи (Animator-инстансы) остаются.
    -- Это НЕ per-killer (Animator может быть от любой модели) — поэтому в scope не идёт.
    for animator, _ in pairs(_animReplacedCaches) do
        if not animator.Parent then _animReplacedCaches[animator] = nil end
    end

    -- 🔪 Slasher-only: глушим beat-visual если уходящий актёр — Slasher.
    -- Если уходящий — другой киллер, stop() уже no-op (модуль не был запущен).
    if SlasherBeatVisual.isMyActor(actorInfo.ActorName) then
        pcall(SlasherBeatVisual.stop)
    end
end

local function startActorListener()
    local ok, Actors = pcall(function() return require(RS:WaitForChild("Modules"):WaitForChild("Gameplay"):WaitForChild("Actors")) end)
    if not ok or not Actors then return end
    _ActorsMod = Actors
    local waited = 0
    while not (Actors.ActorAdded and Actors.ActorRemoved) and waited < 5 do task.wait(0.1); waited = waited + 0.1 end
    if not (Actors.ActorAdded and Actors.ActorRemoved) then return end

    if Actors.CurrentActors then
        for _, info in pairs(Actors.CurrentActors) do
            if info.ActorType == "Killer" then task.spawn(onKillerActorAdded, info) end
        end
    end

    track(Actors.ActorAdded:Connect(function(info) if info.ActorType == "Killer" then task.spawn(onKillerActorAdded, info) end end))
    track(Actors.ActorRemoved:Connect(function(info)
        if info.ActorType == "Killer" then task.spawn(onKillerActorRemoved, info) end
        if info.ActorType == "Survivor" then checkLMSGlobal() end
    end))

    listenEvent("CreateRagdoll", function(victimModel, profile)
        if not victimModel then return end
        local victimPlayer = Players:GetPlayerFromCharacter(victimModel)
        if not (Actors.CurrentActors and victimPlayer) then task.delay(0.1, checkLMSGlobal); return end
        local info = Actors.CurrentActors[victimPlayer]
        if info and info.ActorType == "Survivor" then task.delay(0.1, checkLMSGlobal) end
    end)

    listenEvent("RoundEnded", function() resetLMSState(); if deactivateLMSSfx then pcall(deactivateLMSSfx) end end)
    listenEvent("ReplicateMapCreation", function() resetLMSState() end)

    local function setupMinionWatch()
        local mf = Workspace:FindFirstChild("Map")
        local function watchIngame(ig)
            if not ig or ig.Name ~= "Ingame" then return end
            local function hookMinion(o)
                if skinsEnabled and o.Name == "PizzaDeliveryRig" then task.spawn(applyMinionSkin, o) end
            end
            for _, o in ipairs(ig:GetChildren()) do hookMinion(o) end
            track(ig.ChildAdded:Connect(hookMinion))
        end
        if mf then
            local ig = mf:FindFirstChild("Ingame")
            if ig then watchIngame(ig) end
            track(mf.ChildAdded:Connect(function(c) if c.Name == "Ingame" then watchIngame(c) end end))
        end
    end
    setupMinionWatch(); setupIntroRigWatch()
end

local function makeDraggable(b)
    local dragging, dragStart, startPos
    b.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            dragging, dragStart, startPos = true, i.Position, b.Position
        end
    end)
    b.InputChanged:Connect(function(i)
        if dragging and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
            local d = i.Position - dragStart
            b.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + d.X, startPos.Y.Scale, startPos.Y.Offset + d.Y)
        end
    end)
    b.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then dragging = false end
    end)
end
